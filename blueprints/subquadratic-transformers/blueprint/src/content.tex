\begin{definition}[Inner product of binary/real vectors]
  \label{def:inner-prod}
  \lean{inner}
  \mathlibok
  For $v, w \in \R^{\ell}$ we write $\inner{v}{w} = \sum_{i=1}^{\ell} v[i]\, w[i]$
  for their (standard) inner product, where $v[i]$ denotes the $i$-th entry of
  $v$. Equivalently, for vectors indexed by a finite type $I$, one may write
  $\inner{v}{w} = \sum_{i \in I} v_i w_i$. For binary vectors
  $v,w \in \{0,1\}^{\ell}$ this counts the coordinates on which both are $1$.
  Recalled only so later nodes can depend on it.
\end{definition}

\begin{definition}[$\ell_2$ norm]
  \label{def:l2norm}
  \lean{EuclideanSpace.norm_eq}
  \mathlibok
  For $v \in \R^{\ell}$, $\norm{v} = \sqrt{\inner{v}{v}}$ is the $\ell_2$ norm.
  We also use the $\ell_1$ norm $\norm{v}_1 = \sum_{i=1}^{\ell} |v[i]|$; for a
  binary vector $\norm{v}_1$ is its number of $1$ entries. Standard.
\end{definition}

\begin{definition}[Kronecker (tensor) product of vectors]
  \label{def:kron}
  For finite index sets $I,J$, and vectors $v : I \to \R$ and $w : J \to \R$,
  the Kronecker product $v \otimes w : I \times J \to \R$ is the vector defined
  entrywise by
  \[
    (v \otimes w)_{(i,j)} = v_i w_j .
  \]
  For vectors in $\R^a$ and $\R^b$ this is the same vector in $\R^{ab}$ whose
  entries are all products of one entry of $v$ and one entry of $w$. We write
  $v^{\otimes q}$ for the $q$-fold Kronecker power $v \otimes \cdots \otimes v$.
  This is the vector-indexed Kronecker product used in the paper.

  For the $q$-fold power, fix the following \emph{recursive product} indexing
  convention, which the Kronecker-power nodes below all use. Define the index
  type recursively by
  \[
    I^{\otimes 0} = \mathrm{Unit},
    \qquad
    I^{\otimes (q+1)} = I \times I^{\otimes q},
  \]
  and the power vector recursively by
  \[
    v^{\otimes 0} = (\_ \mapsto 1) : \mathrm{Unit}\to\R,
    \qquad
    v^{\otimes (q+1)} = v \otimes v^{\otimes q}
      = \operatorname{def\_kron}(v, v^{\otimes q}).
  \]
  Both $\mathrm{Unit}$ and $I\times I^{\otimes q}$ are finite types, so
  $v^{\otimes q}$ is a genuine finite-dimensional vector, and the empty product
  gives $v^{\otimes 0}\equiv 1$. This convention makes each power the binary
  Kronecker product of $v$ with the previous power \emph{definitionally}, so the
  induction proofs below reduce to one application of
  Lemma~\ref{lem:kron-inner} with no tuple-reindexing step. Do \emph{not}
  formalize the power with a $\mathrm{Fin}\,q\to I$ (function-tuple) index type:
  that forces a peeling lemma proved via $\mathrm{Fin.cons}$,
  $\mathrm{Fin.consEquiv}$ or $\operatorname{Equiv.sum\_comp}$, which pulls in
  the module `Mathlib.Logic.Equiv.Fin` (unavailable in this build) and whose
  $\mathrm{congr}$-based proof leaves a definitionally closed goal that makes a
  trailing $\mathrm{Finset.prod\_congr}$ fail with ``No goals to be solved''.
\end{definition}

\begin{lemma}[Finite sum over a product type]
  \label{lem:sum-prod-type}
  \lean{Fintype.sum_prod_type}
  \mathlibok
  For finite types $I,J$ and a function $h : I\times J \to \R$,
  \[
    \sum_{p\in I\times J} h(p)
    =
    \sum_{i\in I}\sum_{j\in J} h(i,j).
  \]
  This is the uncurried product-index expansion. In Lean, this is the
  unprimed theorem `Fintype.sum_prod_type`; the primed theorem is for a curried
  function and should not be used when the target is already a sum over
  $p : I\times J$.
\end{lemma}

\begin{lemma}[Product of two finite sums]
  \label{lem:sum-mul-sum}
  \lean{Fintype.sum_mul_sum}
  \mathlibok
  For finite types $I,J$ and functions $f : I\to\R$ and $g : J\to\R$,
  \[
    \left(\sum_{i\in I} f(i)\right)
    \left(\sum_{j\in J} g(j)\right)
    =
    \sum_{i\in I}\sum_{j\in J} f(i)g(j).
  \]
  Equivalently, the displayed double sum factors as the product of the two
  finite sums by using `Fintype.sum_mul_sum` in the reverse direction.
\end{lemma}

\begin{lemma}[Multiplicativity of inner products under Kronecker product]
  \label{lem:kron-inner}
  \uses{def:inner-prod, def:kron, def:l2norm, lem:sum-prod-type, lem:sum-mul-sum}
  For $v,w \in \R^{a}$ and $x,y\in\R^{b}$,
  \[
    \inner{v\otimes x}{w\otimes y}
    =
    \inner{v}{w}\cdot\inner{x}{y}.
  \]
  This node is only the binary Kronecker identity. The $q$-fold consequences
  used in Appendix B are separate nodes
  Lemma~\ref{lem:kron-power-inner}, Lemma~\ref{lem:kron-power-norm}, and
  Lemma~\ref{lem:kron-power-cosine}. A formalization should not bundle those
  conclusions into this theorem; doing so creates a proof obligation whose
  induction and indexing data are absent from the binary statement.
\end{lemma}
\begin{proof}
  \uses{def:inner-prod, def:kron, def:l2norm, lem:sum-prod-type, lem:sum-mul-sum}
  Expanding the definitions over finite index sets $I,J$ gives a sum indexed by
  the product type:
  \[
    \inner{v\otimes x}{w\otimes y}
    = \sum_{p\in I\times J} (v_{p_1} x_{p_2})(w_{p_1} y_{p_2}).
  \]
  First commute and reassociate the scalar factors in each summand:
  \[
    (v_{p_1}x_{p_2})(w_{p_1}y_{p_2})
    =
    (v_{p_1}w_{p_1})(x_{p_2}y_{p_2}).
  \]
  Then use Lemma~\ref{lem:sum-prod-type} with
  $h(i,j)=(v_iw_i)(x_jy_j)$ to rewrite the product-indexed sum as
  \[
    \sum_{i\in I}\sum_{j\in J} (v_i w_i)(x_j y_j).
  \]
  Finally apply Lemma~\ref{lem:sum-mul-sum} in the reverse direction, with
  $f(i)=v_iw_i$ and $g(j)=x_jy_j$, to factor this double sum as
  \[
    \left(\sum_{i\in I} v_i w_i\right)
    \left(\sum_{j\in J} x_j y_j\right)
    = \inner{v}{w}\inner{x}{y}.
  \]
  Iterating this identity proves the statement for $q$-fold Kronecker powers.
  Applying the same identity with $w=v$ and $y=x$, and then taking square roots,
  gives $\norm{v\otimes x}=\norm{v}\norm{x}$ for nonnegative Euclidean norms;
  iterating gives $\norm{v^{\otimes q}}=\norm{v}^{q}$.
\end{proof}

\begin{lemma}[Inner products of Kronecker powers]
  \label{lem:kron-power-inner}
  \uses{def:inner-prod, def:kron, lem:kron-inner}
  For every finite index set $I$, every $v,w:I\to\R$, and every natural
  $q$, with $v^{\otimes q}$ and $w^{\otimes q}$ indexed by the recursive product
  type $I^{\otimes q}$ of Definition~\ref{def:kron},
  \[
    \inner{v^{\otimes q}}{w^{\otimes q}}=\inner{v}{w}^{q}.
  \]
  The formal statement must use the recursive product index type
  ($I^{\otimes 0}=\mathrm{Unit}$, $I^{\otimes(q+1)}=I\times I^{\otimes q}$) fixed
  in Definition~\ref{def:kron}, with the empty product equal to $1$ when $q=0$.
  It must \emph{not} use a $\mathrm{Fin}\,q\to I$ function-tuple index type, and
  its proof must not introduce a per-coordinate peeling lemma.
\end{lemma}
\begin{proof}
  \uses{def:inner-prod, def:kron, lem:kron-inner}
  Induct on $q$.  For $q=0$ the index type is $\mathrm{Unit}$ and both power
  vectors are the constant $1$, so the sum $\inner{v^{\otimes 0}}{w^{\otimes 0}}$
  is $1\cdot 1=1$, matching the scalar power $\inner{v}{w}^{0}=1$.

  For the induction step, the recursive convention gives, definitionally,
  \[
    v^{\otimes(q+1)} = v\otimes v^{\otimes q},
    \qquad
    w^{\otimes(q+1)} = w\otimes w^{\otimes q},
  \]
  as vectors indexed by $I\times I^{\otimes q}$ (no separate reindexing or
  peeling step is required, because this is the definition of the power).
  Lemma~\ref{lem:kron-inner}, applied with the second factor being the
  $q$-fold powers over the index type $I^{\otimes q}$, gives
  \[
    \inner{v^{\otimes(q+1)}}{w^{\otimes(q+1)}}
    =
    \inner{v}{w}\cdot
    \inner{v^{\otimes q}}{w^{\otimes q}}.
  \]
  Substitute the induction hypothesis and simplify scalar powers to obtain
  $\inner{v}{w}^{q+1}$.

  In Lean, define the power by structural recursion on $q$ so that
  $v^{\otimes(q+1)}$ is \emph{syntactically} $\operatorname{def\_kron}(v,
  v^{\otimes q})$, and prove the lemma by `induction q` whose successor case is
  closed by rewriting with Lemma~\ref{lem:kron-inner}, the induction hypothesis
  and `pow_succ`. Do not use `Fin.cons`, `Fin.consEquiv`,
  `Equiv.sum_comp`, `Fintype.sum_equiv` over a `Fin`-tuple equivalence, or any
  lemma from `Mathlib.Logic.Equiv.Fin` or `Mathlib.Algebra.BigOperators.Ring`
  (both modules are unavailable here), and do not follow a `congr`/`rfl`-closed
  goal with a further `Finset.prod_congr` (that yields ``No goals to be
  solved'').
\end{proof}

\begin{lemma}[Euclidean norms of Kronecker powers]
  \label{lem:kron-power-norm}
  \uses{def:l2norm, lem:kron-power-inner}
  For every finite index set $I$, every vector $v:I\to\R$, and every natural
  $q$,
  \[
    \norm{v^{\otimes q}}=\norm{v}^{q}.
  \]
  The proof must use nonnegativity of Euclidean norms and the square identity
  from Lemma~\ref{lem:kron-power-inner}; it is not a purely ring-theoretic
  consequence.
\end{lemma}
\begin{proof}
  \uses{def:l2norm, lem:kron-power-inner}
  Apply Lemma~\ref{lem:kron-power-inner} with $w=v$:
  \[
    \inner{v^{\otimes q}}{v^{\otimes q}}=\inner{v}{v}^{q}.
  \]
  By Definition~\ref{def:l2norm}, the left-hand side is
  $\norm{v^{\otimes q}}^{2}$ and $\inner{v}{v}=\norm{v}^{2}$. Hence
  \[
    \norm{v^{\otimes q}}^{2}=(\norm{v}^{2})^{q}=(\norm{v}^{q})^{2}.
  \]
  Both $\norm{v^{\otimes q}}$ and $\norm{v}^{q}$ are nonnegative, so equality of
  their squares implies equality of the norms.
\end{proof}

\begin{lemma}[Cosine similarity of Kronecker powers]
  \label{lem:kron-power-cosine}
  \uses{def:cosine, lem:kron-power-inner, lem:kron-power-norm}
  For nonzero vectors $v,w\in\R^{\ell}$ and every natural $q$,
  \[
    \cossim{v^{\otimes q}}{w^{\otimes q}}
    =
    \left(\cossim{v}{w}\right)^{q}.
  \]
  This is the exact identity needed in Lemma~\ref{lem:msd-tensor-reduction}.
\end{lemma}
\begin{proof}
  \uses{def:cosine, lem:kron-power-inner, lem:kron-power-norm}
  Expand cosine similarity and apply Lemma~\ref{lem:kron-power-inner} to the
  numerator and Lemma~\ref{lem:kron-power-norm} to both denominator factors:
  \[
    \cossim{v^{\otimes q}}{w^{\otimes q}}
    =
    \frac{\inner{v}{w}^{q}}{\norm{v}^{q}\norm{w}^{q}}.
  \]
  Since $v$ and $w$ are nonzero, their norms are nonzero. Rearranging powers in
  the field gives
  \[
    \frac{\inner{v}{w}^{q}}{\norm{v}^{q}\norm{w}^{q}}
    =
    \left(\frac{\inner{v}{w}}{\norm{v}\norm{w}}\right)^q
    =
    \left(\cossim{v}{w}\right)^q.
  \]
\end{proof}

\begin{lemma}[Continuity is closed under sums, products and composition]
  \label{lem:continuity-closure}
  \lean{Continuous.comp}
  \mathlibok
  This leaf packages standard continuity declarations such as
  `Continuous.add`, `Continuous.sub`, `Continuous.mul`, `Continuous.comp`,
  `continuous_const`, and continuity of coordinate projections. Finite sums,
  finite products and compositions of continuous functions between
  finite-dimensional real vector spaces are continuous, as are constant maps,
  coordinate projections and affine scalar maps. Recalled so the
  piecewise-linear MLP constructions below can be shown continuous.
\end{lemma}

\begin{lemma}[Continuity of max and min with a continuous argument]
  \label{lem:continuous-real-max-min}
  \lean{Continuous.max, Continuous.min}
  \mathlibok
  If $h : X \to \R$ is continuous and $c\in\R$ is constant, then the maps
  $x\mapsto \max(c,h(x))$ and $x\mapsto \min(c,h(x))$ are continuous. In Lean
  these are not proved by treating `max c` or `min c` as independently
  continuous curried functions; apply
  `Continuous.max continuous_const hh` and
  `Continuous.min continuous_const hh` to the full lambda expression.
\end{lemma}

\begin{lemma}[Continuity of the affine divided map]
  \label{lem:continuous-affine-divided}
  \uses{lem:continuity-closure}
  For real numbers $a,b$ with $b>a$, the map
  \[
    y\mapsto \frac{y-a}{b-a}
  \]
  is continuous as a map $\R\to\R$.
\end{lemma}
\begin{proof}
  \uses{lem:continuity-closure}
  Since $b>a$, the denominator $b-a$ is nonzero. Rewrite division by the
  constant $b-a$ as multiplication by the constant $(b-a)^{-1}$:
  \[
    \frac{y-a}{b-a}=(y-a)(b-a)^{-1}.
  \]
  The map $y\mapsto y-a$ is continuous by continuity of the identity map,
  constants and subtraction, and multiplication by the constant $(b-a)^{-1}$
  preserves continuity. Thus the displayed affine-divided map is continuous.

  A Lean proof should expose this as a separate continuous term, for example:
  first prove `hden : b - a \ne 0`; then prove
  `h_aff : Continuous (fun y : \R => (y - a) / (b - a))` by rewriting
  `div_eq_mul_inv` and using continuity of subtraction and multiplication by a
  constant. This proof should not be hidden inside a later `continuity` call.
\end{proof}

\begin{lemma}[Continuity of finite coordinate products]
  \label{lem:continuous-finite-product}
  \lean{continuous_finsetProd}
  \mathlibok
  Let $I$ be finite and let $h_i : X\to\R$ be continuous for every $i\in I$.
  Then $x\mapsto \prod_{i\in I} h_i(x)$ is continuous. In Lean, for a product
  over all coordinates $i:\mathrm{Fin}\,\ell$, use
  `continuous_finsetProd Finset.univ` and prove each factor continuous.
\end{lemma}

\begin{lemma}[Continuity of the clipped affine threshold]
  \label{lem:continuous-clipped-affine}
  \uses{lem:continuous-affine-divided, lem:continuous-real-max-min}
  For real numbers $a,b$ with $b>a$, the function
  \[
    g(y)=\max\!\left(0,\min\!\left(1,\frac{y-a}{b-a}\right)\right)
  \]
  is continuous as a map $\R\to\R$.
\end{lemma}
\begin{proof}
  \uses{lem:continuous-affine-divided, lem:continuous-real-max-min}
  By Lemma~\ref{lem:continuous-affine-divided}, the inner affine-divided map
  \[
    h(y)=\frac{y-a}{b-a}
  \]
  is continuous. Apply Lemma~\ref{lem:continuous-real-max-min} to the continuous
  map $h$ and the constant $1$ to obtain continuity of
  \[
    y\mapsto \min(1,h(y)).
  \]
  Call this continuous map $h_{\min}$. Applying
  Lemma~\ref{lem:continuous-real-max-min} again, now to $h_{\min}$ and the
  constant $0$, gives continuity of
  \[
    y\mapsto \max(0,h_{\min}(y))
    =
    \max\!\left(0,\min\!\left(1,\frac{y-a}{b-a}\right)\right).
  \]

  For a Lean proof, do not use the tactic `continuity` for this node, because it
  may create unprovable curried goals such as `Continuous (max 0)` or
  `Continuous (min 1)`. The intended proof is the direct one:
  define
  `h_aff : Continuous (fun y : \R => (y - a) / (b - a))` using
  Lemma~\ref{lem:continuous-affine-divided}; then set
  `h_min : Continuous (fun y : \R => min 1 ((y - a) / (b - a))) :=
  Continuous.min continuous_const h_aff`; finally conclude with
  `Continuous.max continuous_const h_min`.
\end{proof}

\begin{lemma}[Values of the clipped affine threshold]
  \label{lem:clipped-affine-values}
  \uses{lem:continuous-clipped-affine}
  For real numbers $a,b$ with $b>a$, define
  \[
    g(y)=\max\!\left(0,\min\!\left(1,\frac{y-a}{b-a}\right)\right).
  \]
  Then
  \[
    y\ge b \Longrightarrow g(y)=1,
    \qquad
    y<a \Longrightarrow g(y)=0.
  \]
  The two endpoint behaviors are part of this node. Continuity alone is not
  enough for Lemma~\ref{lem:mlp-threshold}.
\end{lemma}
\begin{proof}
  \uses{lem:continuous-clipped-affine}
  Since $b>a$, the denominator $b-a$ is strictly positive. If $y\ge b$, then
  $y-a\ge b-a$, so $(y-a)/(b-a)\ge1$. Hence the inner minimum is $1$, and the
  outer maximum is also $1$.

  If $y<a$, then $y-a<0$, so $(y-a)/(b-a)<0$. Hence the inner minimum is at
  most this negative number, and the outer maximum with $0$ is $0$.
\end{proof}

\begin{lemma}[Finite products with one zero factor and all-one factors]
  \label{lem:finite-product-zero-one}
  \lean{Finset.prod_eq_zero, Finset.prod_congr}
  \mathlibok
  Let $I$ be finite and $h:I\to\R$. If $h(i_0)=0$ for some $i_0$, then
  $\prod_{i\in I}h(i)=0$. If $h(i)=1$ for every $i$, then
  $\prod_{i\in I}h(i)=1$. These are standard finite-product facts used to turn
  the coordinatewise clipped threshold values into the behavior of the product
  MLP.
\end{lemma}

\begin{lemma}[Scalar division by a product]
  \label{lem:scalar-div-product}
  \lean{div_eq_mul_inv}
  \mathlibok
  For real scalars $s,\alpha,\beta$ with $\alpha\ne0$ and $\beta\ne0$,
  \[
    s\cdot(\alpha^{-1}\beta^{-1})=\frac{s}{\alpha\beta}.
  \]
  This is the elementary field simplification obtained from
  $(\alpha\beta)^{-1}=\alpha^{-1}\beta^{-1}$. In Lean, the corresponding scalar
  goal is closed directly by `field_simp [hα, hβ]` using the nonzero
  hypotheses; there is no subsequent `ring`/`nlinarith` step, because the
  simplifier has already closed the goal in this scalar form.
\end{lemma}

\begin{lemma}[Scaled summand factorization]
  \label{lem:scaled-summand-factor}
  For real scalars $x,y,\alpha,\beta$,
  \[
    (x/\alpha)(y/\beta) = (xy)(\alpha^{-1}\beta^{-1}).
  \]
  This is only the definition of division as multiplication by inverse, followed
  by commutativity and associativity of multiplication. No nonzero denominator
  hypotheses are required for this factorization.
\end{lemma}
\begin{proof}
  Rewrite $x/\alpha$ as $x\alpha^{-1}$ and $y/\beta$ as
  $y\beta^{-1}$. Then reassociate and commute the four scalar factors:
  \[
    (x\alpha^{-1})(y\beta^{-1})
    = x y \alpha^{-1}\beta^{-1}
    = (xy)(\alpha^{-1}\beta^{-1}).
  \]
  In Lean this node should be proved by unfolding division to multiplication by
  inverse and normalizing the commutative monoid product; it should not use
  `field_simp`, because there are no denominator hypotheses and no field
  cancellation is involved.
\end{proof}

\begin{lemma}[Scaling both arguments of the finite-sum inner product]
  \label{lem:inner-scaled}
  \uses{def:inner-prod, lem:scaled-summand-factor, lem:scalar-div-product}
  For any finite index set $I$, vectors $v,w : I \to \R$, and scalars
  $\alpha,\beta \in \R$ with $\alpha\ne 0$ and $\beta\ne 0$,
  \[
    \inner{v/\alpha}{w/\beta}
    =
    \frac{\inner{v}{w}}{\alpha\beta}.
  \]
  Equivalently, after unfolding the inner product as a finite sum,
  \[
    \alpha\beta
    \sum_{i\in I} v_i w_i \alpha^{-1}\beta^{-1}
    =
    \sum_{i\in I} v_i w_i .
  \]
\end{lemma}
\begin{proof}
  \uses{def:inner-prod, lem:scaled-summand-factor, lem:scalar-div-product}
  Expanding the left-hand inner product gives
  \[
    \inner{v/\alpha}{w/\beta}
    =
    \sum_{i\in I} (v_i\alpha^{-1})(w_i\beta^{-1}).
  \]
  For each index $i$, apply Lemma~\ref{lem:scaled-summand-factor}, with
  $x=v_i$ and $y=w_i$, to rewrite the summand as
  \[
    (v_i\alpha^{-1})(w_i\beta^{-1})
    =
    (v_iw_i)(\alpha^{-1}\beta^{-1}).
  \]
  This is a pure reassociation/commutation step and has no nonzero hypotheses;
  in particular the formal proof of this step should use
  Lemma~\ref{lem:scaled-summand-factor}, not a `field_simp` step followed by an
  additional normalizer.
  Since $\alpha^{-1}\beta^{-1}$ is independent of $i$, the finite-sum
  constant-multiple rule gives
  \[
    \sum_{i\in I} (v_iw_i)(\alpha^{-1}\beta^{-1})
    =
    \left(\sum_{i\in I} v_iw_i\right)(\alpha^{-1}\beta^{-1}).
  \]
  Apply Lemma~\ref{lem:scalar-div-product} with
  $s=\sum_{i\in I}v_iw_i$. This rewrites the last display directly as
  $\inner{v}{w}/(\alpha\beta)$ and proves the desired equality. The multiplied
  finite-sum identity displayed in the statement is only the same equality after
  multiplying by the nonzero scalar $\alpha\beta$; it is not an additional step
  needed for the main proof.
\end{proof}

\begin{definition}[Truly subquadratic time]
  \label{def:truly-subquadratic}
  An algorithm for a problem on inputs of size parameter $n$ comes with a
  concrete worst-case runtime function $T:\N\to\R_{\ge 0}$ (or
  $T:\N\to\N$ embedded in $\R_{\ge0}$). It runs in \emph{truly subquadratic
  time} iff there exist constants $\eps>0$, $C>0$ and $n_0$ such that for every
  $n\ge n_0$,
  \[
    T(n) \le C n^{2-\eps}.
  \]
  For problems with secondary parameters such as $\ell,t,\gamma$, the runtime
  predicate also records the parameter regime and any permitted
  $\operatorname{poly}(\ell)$ factor; the exponent $\eps$ is fixed independently
  of $n$ and the instance. A formal definition that only asserts the existence
  of a positive $\eps$ and does not mention a runtime function $T$ is vacuous and
  is not this node.
\end{definition}

\begin{definition}[Concrete problem families]
  \label{def:problem-family}
  This node introduces the auditable structure
  $\operatorname{ConcreteProblemFamily}$. The declaration checked at this label
  must itself expose the concrete fields below. It may be a Lean `structure', an
  equivalent dependent Sigma package, or a reducible alias/re-export only if
  unfolding the declaration at this exact label exposes the same projections and
  field types. It is not enough for this node to be an abbreviation such as
  \[
    \operatorname{def\_problem\_family} : \mathrm{Type}\;1
      := \operatorname{ConcreteProblemFamily}
  \]
  when the fields of $\operatorname{ConcreteProblemFamily}$ are not visible at
  the audited declaration.

  A \emph{problem family} $\mathcal P$ is concrete data, not a name tag. An
  element $P:\operatorname{ConcreteProblemFamily}$ has:
  \begin{itemize}
    \item a parameter type
      $P.\operatorname{Param}:\mathrm{Type}$;
    \item an input type family
      $P.\operatorname{Inst}:P.\operatorname{Param}\to\mathrm{Type}$;
    \item a kind field
      $P.\operatorname{Kind}\in
      \{\mathrm{search},\mathrm{decision},\mathrm{promise}\}$;
    \item a search-output type family
      $P.\operatorname{Out}:\prod_p
      P.\operatorname{Inst}(p)\to\mathrm{Type}$;
    \item a search-output correctness relation
      $P.\operatorname{ValidOut}:\prod_{p,I}
      P.\operatorname{Out}(p,I)\to\mathrm{Prop}$;
    \item a yes predicate
      $P.\operatorname{Yes}:\prod_p
      P.\operatorname{Inst}(p)\to\mathrm{Prop}$;
    \item a no predicate
      $P.\operatorname{No}:\prod_p
      P.\operatorname{Inst}(p)\to\mathrm{Prop}$.
  \end{itemize}
  For search families, $\operatorname{Out}$ and $\operatorname{ValidOut}$ are
  the nontrivial fields; for decision families, $\operatorname{Yes}$ is the
  nontrivial field; for promise families, both $\operatorname{Yes}$ and
  $\operatorname{No}$ are nontrivial. Unused fields may be filled with a
  canonical unit-valued default, but the projections must still exist with the
  displayed names and types so that later algorithm and reduction packages can
  select their outputs and correctness obligations by $P.\operatorname{Kind}$.

  The parameter tuple is part of the mathematical object: changing $\ell$, $t$
  or $\gamma$ changes the family. For concrete families in this blueprint the
  parameter value $p$ must contain every displayed parameter, such as
  $(n,\ell)$, $(n,\ell,t)$, $(n,\ell,\gamma)$, $(n,\ell,\alpha,\gamma)$, and any
  bichromatic color-class distinction. Subsequent nodes such as $\MSD$, $\OV$
  and $\MaxIP$ must provide unfolding theorems or projection equations
  identifying these fields with their stated input types and relations.

  A formalization that replaces a concrete family by `Unit', a string, an enum,
  a bare tag, an opaque abbreviation, or a declaration that ends in `True' has
  erased the problem and does not satisfy this node. In particular, if the
  generated Lean declares only
  `abbrev def_problem_family : Type 1 := ConcreteProblemFamily', then the
  statement alignment audit must reject it unless this very declaration also
  unfolds to the fields above.
\end{definition}

\begin{definition}[No placeholder problem families]
  \label{def:parameter-sensitive-placeholder}
  \uses{def:problem-family}
  This node introduces, or transparently re-exports, the auditable structure
  $\operatorname{ConcreteProblemFamily}$ used by every reusable problem-family
  wrapper in the blueprint. The mathematical object at this label is a
  structure with the fields below. It may be implemented directly as a
  structure declaration, or as a reducible alias/re-export of an already defined
  structure with exactly these projections; in the alias case, unfolding the
  declaration audited at this label must expose the same field names and field
  types. It is not a marker proposition and it is not an opaque family tag. An
  element $P:\operatorname{ConcreteProblemFamily}$ consists of the following
  fields:
  \[
  \begin{array}{ll}
    P.\operatorname{Param} &: \mathrm{Type},\\
    P.\operatorname{Inst} &: P.\operatorname{Param}\to\mathrm{Type},\\
    P.\operatorname{Kind} &: \{\mathrm{search},\mathrm{decision},\mathrm{promise}\},\\
    P.\operatorname{Out} &: \prod_p P.\operatorname{Inst}(p)\to\mathrm{Type}
      \quad\text{for search families},\\
    P.\operatorname{ValidOut} &: \prod_{p,I}
      P.\operatorname{Out}(p,I)\to\mathrm{Prop}
      \quad\text{for search families},\\
    P.\operatorname{Yes} &: \prod_p P.\operatorname{Inst}(p)\to\mathrm{Prop}
      \quad\text{for decision families},\\
    P.\operatorname{No} &: \prod_p P.\operatorname{Inst}(p)\to\mathrm{Prop}
      \quad\text{for promise families}.
  \end{array}
  \]
  Search families have a nontrivial $\operatorname{Out}$ and
  $\operatorname{ValidOut}$ field; decision families have a nontrivial
  $\operatorname{Yes}$ field; promise families have both nontrivial
  $\operatorname{Yes}$ and $\operatorname{No}$ fields. Unused fields may be
  filled by a canonical unit-valued default, but the kind-specific fields just
  listed are mandatory data of the structure.

  The parameter value $p$ must include every displayed parameter of the problem,
  including $\ell$, $t$, $\gamma$, $\alpha$, and any color-class distinction.
  For each named family below there must be an unfolding theorem identifying
  these fields with the displayed mathematics. For example, unfolding
  the concrete family for $\MSD_{n,\ell,t}$ must give exactly
  \[
    P.\operatorname{Inst}(n,\ell,t)
      =\{(v_1,\ldots,v_n): v_i\in\{0,1\}^{\ell},\ v_i\ne0\},
    \qquad
    P.\operatorname{Yes}(n,\ell,t,I)\Longleftrightarrow
    \exists\, i\ne j,\ \cossim{v_i}{v_j}\ge t.
  \]
  Similarly, unfolding $\gamma$-Min-IP must expose the approximation factor
  $\gamma$ and its two approximation inequalities, and unfolding an additive
  promise problem must expose both its yes and no predicates. A declaration at
  this label whose body is a non-reducible abbreviation, whose target structure
  lacks the displayed projections, whose statement is a bare proposition, which
  ends in `True', which uses `Unit' as the family, or which stores only a
  name/tag/enum plus parameters does not satisfy this node. A transparent alias
  to a previously declared structure is acceptable precisely when field access
  through the audited name reduces to the displayed fields above.
\end{definition}

\begin{definition}[Truly subquadratic algorithm predicate]
  \label{def:tsq-alg-exists}
  \uses{def:problem-family, def:parameter-sensitive-placeholder, def:truly-subquadratic}
  For a concrete problem family $\mathcal P$, this node introduces, or
  transparently re-exports, the auditable structure named at this label. In the
  generated Lean this declaration is `def_tsq_alg_exists'. The declaration
  `def_tsq_alg_exists P' itself must be the algorithm package, an equivalent
  dependent Sigma package with the same projections, or a reducible alias to
  such a package. If a helper structure named
  $\operatorname{TSQAlgorithm}(\mathcal P)$ is also used, then
  `def_tsq_alg_exists P' must be definitionally equal to it, for example a
  transparent abbreviation
  \[
    \operatorname{def\_tsq\_alg\_exists}(\mathcal P)
      := \operatorname{TSQAlgorithm}(\mathcal P).
  \]
  It is not merely the proposition
  $\operatorname{Nonempty}(\operatorname{TSQAlgorithm}(\mathcal P))$ or
  $\operatorname{Nonempty}(\operatorname{def\_tsq\_alg\_exists}(\mathcal P))$.
  If implemented as an alias, unfolding the audited declaration
  `def_tsq_alg_exists' must expose the regime, run, correctness, runtime and
  exponent fields below. An element
  $\mathcal A:\operatorname{def\_tsq\_alg\_exists}(\mathcal P)$, equivalently
  $\mathcal A:\operatorname{TSQAlgorithm}(\mathcal P)$ when that name is a
  transparent alias, has the following fields:
  \begin{itemize}
    \item a parameter-regime predicate
      $\operatorname{Regime}_{\mathcal A} :
      \operatorname{Param}_{\mathcal P}\to\mathrm{Prop}$ specifying the
      asymptotic slice under discussion, such as
      $\ell=\logDim(c,n)$ or $\ell=\msdHardDim(c,n)$;
    \item a single dependent run value, whose type is selected by
      $\mathcal P.\operatorname{Kind}$:
      \[
      \operatorname{AlgOut}_{\mathcal P}(p,I) :=
      \begin{cases}
        \operatorname{Out}_{\mathcal P}(p,I)
          & \text{if }\mathcal P.\operatorname{Kind}=\mathrm{search},\\
        \{\mathrm{true},\mathrm{false}\}
          & \text{if }\mathcal P.\operatorname{Kind}=\mathrm{decision},\\
        \{\mathrm{true},\mathrm{false}\}
          & \text{if }\mathcal P.\operatorname{Kind}=\mathrm{promise}.
      \end{cases}
      \]
      The field is
      \[
        \operatorname{run}_{\mathcal A}(p,I):
        \operatorname{AlgOut}_{\mathcal P}(p,I)
      \]
      for every in-regime $p$ and every instance $I$. There are not separate
      fields called search-run, decision-run and promise-run; requiring all of
      them for every family changes the meaning of the algorithm package.
    \item a single dependent correctness proof, again selected by
      $\mathcal P.\operatorname{Kind}$:
      \[
      \operatorname{AlgCorrect}_{\mathcal P}(p,I,o) :=
      \begin{cases}
        \operatorname{ValidOut}_{\mathcal P}(p,I,o)
          & \text{if }\mathcal P.\operatorname{Kind}=\mathrm{search},\\
        \bigl(o=\mathrm{true}\Longleftrightarrow
          \operatorname{Yes}_{\mathcal P}(p,I)\bigr)
          & \text{if }\mathcal P.\operatorname{Kind}=\mathrm{decision},\\
        \bigl(\operatorname{Yes}_{\mathcal P}(p,I)\Rightarrow
          o=\mathrm{true}\bigr)\wedge
        \bigl(\operatorname{No}_{\mathcal P}(p,I)\Rightarrow
          o=\mathrm{false}\bigr)
          & \text{if }\mathcal P.\operatorname{Kind}=\mathrm{promise}.
      \end{cases}
      \]
      The field is
      \[
        \operatorname{correct}_{\mathcal A}(p,I):
        \operatorname{AlgCorrect}_{\mathcal P}
          (p,I,\operatorname{run}_{\mathcal A}(p,I)).
      \]
      Thus a search algorithm proves only the search output relation; it does
      not also return a Boolean or prove a decision/promise condition. A
      decision or promise algorithm returns only a Boolean; it does not also
      construct a search output. Implementing this node with all three runners
      and all three correctness obligations for every $\mathcal P$ is rejected
      as a different, stronger and semantically misaligned structure.
    \item a worst-case runtime function
      \[
        T_{\mathcal A}:\operatorname{Param}_{\mathcal P}\to\R_{\ge0}
      \]
      bounding the run time on all instances with that parameter;
    \item constants $\eps>0$, $C>0$ and $n_0$ proving that
      $T_{\mathcal A}$ satisfies Definition~\ref{def:truly-subquadratic} on
      $\operatorname{Regime}_{\mathcal A}$, including any allowed
      $\operatorname{poly}(\ell)$ secondary factor and with $\eps$ independent
      of $n$.
  \end{itemize}
  After this structure is defined, the notation
  \[
    \TSQAlg{\mathcal P}
    \quad\text{means}\quad
    \operatorname{Nonempty}(
      \operatorname{def\_tsq\_alg\_exists}(\mathcal P)).
  \]
  If the implementation also exposes the notation as
  $\operatorname{Nonempty}(\operatorname{TSQAlgorithm}(\mathcal P))$, then the
  transparent equality
  $\operatorname{def\_tsq\_alg\_exists}(\mathcal P)
  = \operatorname{TSQAlgorithm}(\mathcal P)$ must be available by reduction at
  this label.
  Thus $\TSQAlg{\MSD_{n,\ell,t}}$ asserts existence of an algorithm for the
  concrete threshold cosine-similarity predicate, not for a family tag named
  ``MSD''. A Lean declaration at this label that merely states
  `Nonempty (TSQAlgorithm P)' is too weak, because the statement being audited
  would not itself expose the regime, run, correctness, runtime function, and
  truly-subquadratic bound data. A reducible type alias to an existing
  `TSQAlgorithm P' structure is acceptable only if these projections are
  available after unfolding `def_tsq_alg_exists'. If an implementation uses helper
  definitions such as `AlgOut' and `AlgCorrect', this node must define them,
  import them transparently, or unfold them at this label, and the structure
  must have exactly one kind-indexed run field and exactly one kind-indexed
  correctness field as above.
\end{definition}

\begin{definition}[Truly subquadratic lower-bound predicate]
  \label{def:no-tsq-alg}
  \uses{def:problem-family, def:truly-subquadratic, def:tsq-alg-exists}
  For a problem family $\mathcal P$, the assertion $\NoTSQAlg{\mathcal P}$ is
  definitionally the direct negation of the concrete algorithm package from
  Definition~\ref{def:tsq-alg-exists}:
  \[
    \NoTSQAlg{\mathcal P}
    \quad := \quad
    \neg\,\operatorname{Nonempty}
      \bigl(\operatorname{def\_tsq\_alg\_exists}(\mathcal P)\bigr).
  \]
  Equivalently, it is the function type
  \[
    \operatorname{def\_tsq\_alg\_exists}(\mathcal P)\to\mathrm{False}
  \]
  after unpacking `Nonempty'. The definition at this label must mention
  `def_tsq_alg_exists P' itself and must reduce to its negation. It may also
  display $\operatorname{TSQAlgorithm}(\mathcal P)$, but only when
  `def_tsq_alg_exists P' is a transparent alias for
  $\operatorname{TSQAlgorithm}(\mathcal P)$ at
  Definition~\ref{def:tsq-alg-exists}. It must not negate
  `Nonempty (TSQAlgorithm P)' while leaving `def_tsq_alg_exists P' as a
  different audited declaration, and it must not negate a separate opaque
  predicate named `TSQAlg P' unless that notation is a transparent abbreviation
  for the displayed `Nonempty (def_tsq_alg_exists P)'. Otherwise the lower-bound
  predicate is no longer audibly tied to the run, correctness, runtime and
  exponent fields specified in Definition~\ref{def:tsq-alg-exists}.
\end{definition}

\begin{definition}[Fine-grained reduction and equivalence predicates]
  \label{def:fg-reduction-predicates}
  \uses{def:problem-family, def:tsq-alg-exists}
  For concrete problem families $\mathcal P$ and $\mathcal Q$, this node
  introduces, or transparently re-exports, the auditable witness structure
  $\operatorname{FGReduction}(\mathcal P,\mathcal Q)$. The declaration at this
  label is the structure itself, an equivalent dependent Sigma package with the
  same projections, or a reducible alias to such a package. It is not merely the
  proposition
  $\operatorname{Nonempty}(\operatorname{FGReduction}(\mathcal P,\mathcal Q))$.
  If implemented as an alias, unfolding the audited declaration must expose the
  regime, call-list, reconstruction, correctness and runtime-accounting fields
  below. An element $R:\operatorname{FGReduction}(\mathcal P,\mathcal Q)$ has
  fields:
  \begin{itemize}
    \item a source regime
      $\operatorname{SrcRegime}_R:
      \operatorname{Param}_{\mathcal P}\to\mathrm{Prop}$ and a target regime
      $\operatorname{TgtRegime}_R:
      \operatorname{Param}_{\mathcal Q}\to\mathrm{Prop}$, together with an
      eventual-coverage theorem for the displayed source slice. Concretely,
      each reduction states a slice predicate
      $\operatorname{SrcSlice}_R$ and proves
      \[
        \exists n_0,\ \forall p,\ n(p)\ge n_0 \wedge
        \operatorname{SrcSlice}_R(p)
        \Longrightarrow \operatorname{SrcRegime}_R(p).
      \]
      For example, on the logarithmic slice this says that all sufficiently
      large $(n,\ell)$ with $\ell=\logDim(c,n)$ are included. Mere nonemptiness
      of $\operatorname{SrcRegime}_R$ is not enough, and
      $\operatorname{SrcRegime}_R=\emptyset$ is forbidden;
    \item a parameter transformation from each in-regime parameter
      $p_{\mathcal P}$ to a parameter $p_{\mathcal Q}$ of $\mathcal Q$;
    \item a finite oracle-call family, not just a single target instance:
      \[
        \operatorname{Calls}_R(p_{\mathcal P},I):
        \operatorname{Finset}\!
        \left(\sum_{p_{\mathcal Q}:\operatorname{Param}_{\mathcal Q}}
          \operatorname{Inst}_{\mathcal Q}(p_{\mathcal Q})\right).
      \]
      Every call in this finite set must satisfy
      $\operatorname{TgtRegime}_R(p_{\mathcal Q})$. Single-call reductions are
      represented by a singleton finite set; recursive split reductions such as
      Lemma~\ref{lem:minip-bichromatic-reduction} use many calls;
    \item an oracle-answer type for each call, defined by the same kind-indexed
      output convention as Definition~\ref{def:tsq-alg-exists}: a valid target
      search output for search families, and a Boolean answer for decision or
      promise families;
    \item an output reconstruction procedure
      \[
        \operatorname{reconstruct}_R(p_{\mathcal P},I,\operatorname{answers})
        :
        \operatorname{AlgOut}_{\mathcal P}(p_{\mathcal P},I),
      \]
      where `answers' assigns an oracle answer to every element of
      $\operatorname{Calls}_R(p_{\mathcal P},I)$;
    \item a correctness theorem saying that if every oracle answer satisfies
      $\operatorname{AlgCorrect}_{\mathcal Q}$ for its concrete target call,
      then the reconstructed answer satisfies
      $\operatorname{AlgCorrect}_{\mathcal P}$ for the original instance. This
      theorem must use the concrete $\operatorname{ValidOut}$,
      $\operatorname{Yes}$ and $\operatorname{No}$ relations exposed by the two
      problem families;
    \item a runtime accounting package, with named projections and defining
      equations. The structure contains a concrete overhead function
      \[
        \operatorname{overhead}_R(p_{\mathcal P},I)\in\R_{\ge0}
      \]
      for forming the call list and reconstructing the source answer. For every
      target algorithm
      $\mathcal A_Q:\operatorname{TSQAlgorithm}(\mathcal Q)$ it defines the
      oracle answers by the equation
      \[
        \operatorname{oracleAnswer}_{R,\mathcal A_Q}(c)
        =
        \mathcal A_Q.\operatorname{run}(p_c,I_c),
        \qquad
        c=(p_c,I_c)\in\operatorname{Calls}_R(p_{\mathcal P},I),
      \]
      and defines the induced source run by the equation
      \[
        \operatorname{inducedRun}_{R,\mathcal A_Q}(p_{\mathcal P},I)
        =
        \operatorname{reconstruct}_R
          (p_{\mathcal P},I,
           \operatorname{oracleAnswer}_{R,\mathcal A_Q}).
      \]
      Its correctness field is exactly the correctness theorem above applied to
      these oracle answers and to
      $\mathcal A_Q.\operatorname{correct}$ for each concrete call. The induced
      source runtime is also defined, not chosen:
      \[
        \operatorname{inducedRuntime}_{R,\mathcal A_Q}(p_{\mathcal P},I)
        =
        C_R\cdot\operatorname{overhead}_R(p_{\mathcal P},I)
        + C_R\sum_{c=(p_{\mathcal Q},I')\in
          \operatorname{Calls}_R(p_{\mathcal P},I)}
          T_{\mathcal A_Q}(p_{\mathcal Q}),
      \]
      for a fixed constant $C_R>0$ belonging to the reduction witness. The
      runtime-bound theorem states that the runtime of the constructed source
      algorithm is bounded by this very
      $\operatorname{inducedRuntime}_{R,\mathcal A_Q}$, and the
      subquadratic-preservation theorem proves that, whenever
      $\mathcal A_Q$ has exponent $\eps>0$ on the target regime, the induced
      runtime is $O(n^{2-\eps'}\operatorname{poly}(\ell))$ for some
      $\eps'>0$ on the covered source slice. This proof must explicitly account
      for the parameter transformation, dimension blowup, number of calls,
      overhead, and all $\operatorname{poly}(\ell)$ factors.
  \end{itemize}
  The induced algorithm is not independent data. In a compliant Lean
  implementation, the reduction witness does \emph{not} contain a free field
  with type
  \[
    \operatorname{transformedAlgorithm} :
      \operatorname{def\_tsq\_alg\_exists}(\mathcal Q)\to
      \operatorname{def\_tsq\_alg\_exists}(\mathcal P),
  \]
  nor the same type with $\operatorname{TSQAlgorithm}$ in place of
  `def_tsq_alg_exists'. Instead, the transformed source algorithm is the
  transparent definition
  \[
    \operatorname{mkInducedAlgorithm}(R,\mathcal A_Q),
  \]
  whose projections reduce to the formulas already displayed:
  \[
  \begin{array}{ll}
    \operatorname{mkInducedAlgorithm}(R,\mathcal A_Q).\operatorname{run}
      &= \operatorname{inducedRun}_{R,\mathcal A_Q},\\
    \operatorname{mkInducedAlgorithm}(R,\mathcal A_Q).\operatorname{correct}
      &= \operatorname{inducedCorrect}_{R,\mathcal A_Q},\\
    \operatorname{mkInducedAlgorithm}(R,\mathcal A_Q).T
      &= \operatorname{inducedRuntime}_{R,\mathcal A_Q},\\
    \operatorname{mkInducedAlgorithm}(R,\mathcal A_Q).\operatorname{eps}
      &>0
        \quad\text{is the exponent produced by the
        subquadratic-preservation theorem.}
  \end{array}
  \]
  If an implementation insists on exposing a convenience name
  `transformedAlgorithm' or `algorithmTransform', that name must be a reducible
  definition or theorem alias for $\operatorname{mkInducedAlgorithm}$, not a
  structure field. At minimum, the audited declaration must include projection
  equations named, for example,
  `transformedAlgorithm_run',
  `transformedAlgorithm_correct',
  `transformedAlgorithm_runtime', and
  `transformedAlgorithm_tsq', proving that the alleged transformed algorithm's
  run, correctness proof, runtime function, and truly-subquadratic bound are
  exactly the induced objects above. A field
  `transformedAlgorithm : TSQAlgorithm Q -> TSQAlgorithm P' with no such
  projection equations is rejected even if the structure separately contains
  call lists, reconstruction maps, or runtime expressions.
  The eventual-coverage and call/reconstruction fields are logically essential:
  choosing $\operatorname{SrcRegime}_R=\emptyset$ makes all later obligations
  vacuous, and restricting reductions to one target instance cannot express the
  recursive reductions used in the paper. Every theorem using this structure
  must specify the concrete asymptotic source slice it reduces, and its proof
  object must act on arbitrary sufficiently large instances of that slice.
  The reduction predicate is then
  \[
    \ReducesTo{\mathcal P}{\mathcal Q}
    \quad\Longleftrightarrow\quad
    \operatorname{Nonempty}(\operatorname{FGReduction}(\mathcal P,\mathcal Q)).
  \]
  Every witness canonically induces the logical implication
  $\TSQAlg{\mathcal Q}\Rightarrow \TSQAlg{\mathcal P}$ by building the
  $\operatorname{TSQAlgorithm}(\mathcal P)$ whose run, correctness and runtime
  fields are respectively
  $\operatorname{inducedRun}_{R,\mathcal A_Q}$, the induced correctness theorem,
  and $\operatorname{inducedRuntime}_{R,\mathcal A_Q}$ above. The audited
  declaration must expose `mkInducedAlgorithm' or an equivalent transparent
  constructor with these exact projection equations. There is no free-standing
  opaque `transformedAlgorithm' or `algorithmTransform' field whose behavior can
  be unrelated to the parameter map, call list, target runs, reconstruction,
  correctness proof and runtime theorem. In particular, a structure containing
  an unconstrained field
  `transformedAlgorithm : TSQAlgorithm Q -> TSQAlgorithm P' fails this node
  unless that field is definitionally forced, by the projection equations above,
  to be the induced algorithm. There is also no arbitrary
  `runtimeExpression' field: a nonnegative expression with no equality to the
  target-runtime sum is not runtime accounting. If helper fields with these
  names are present, their equations must reduce to
  $\operatorname{inducedRun}_{R,\mathcal A_Q}$ and
  $\operatorname{inducedRuntime}_{R,\mathcal A_Q}$ as displayed above. The
  implication alone is not the definition of reduction in this blueprint; it
  erases the transformed algorithm, correctness proof, and runtime accounting.
  We write
  $\SubquadEquiv{\mathcal P}{\mathcal Q}$ for the conjunction of the two
  reductions, and $\AtLeastHard{\mathcal Q}{\mathcal P}$ for
  $\ReducesTo{\mathcal P}{\mathcal Q}$. A Lean declaration at this label that
  merely states `Nonempty (FGReduction P Q)' is too weak unless this label is
  also the declaration exposing the fields above. A reducible type alias to an
  existing `FGReduction P Q' package is acceptable only if the projections and
  equations above are available after unfolding the alias. The node is not
  `Unit', not an uninterpreted relation, and not merely the formula
  $\TSQAlg{\mathcal Q}\to\TSQAlg{\mathcal P}$.
\end{definition}

\begin{definition}[Named logarithmic parameter regimes]
  \label{def:log-parameter-regimes}
  \uses{def:problem-family}
  The paper repeatedly uses the same logarithmic dimension and approximation
  regimes. This node must introduce concrete definitions for the six named
  logarithmic-regime functions and predicate. The declaration audited at
  label \ref{def:log-parameter-regimes} is the package that owns these six
  objects. It is not allowed to be just a value that repackages previously
  existing root constants, and it is not allowed to be a structure containing
  only formula proofs about identifiers defined elsewhere.

  A compliant Lean implementation has one of the following two shapes, and in
  both shapes the declaration audited at this label has data fields for the six
  functions/predicate:
  \begin{itemize}
    \item six transparent top-level declarations named exactly
      `logn', `logDim', `polylogDim', `msdHardDim', `mildMsdGamma', and
      `msdHardGammaAllowed', each with the defining equation displayed below.
      The label declaration must then be a reducible package, such as
      `def_log_parameter_regimes : LogParameterRegimePackage', whose data fields
      are exactly those six declarations and whose proof fields expose the six
      equations for those data fields. The package may set, for example,
      `logn := logn', but the field `logn' must still be present in the audited
      package; or
    \item a single structure/Sigma value named for this label whose data fields
      are the six functions/predicate and whose proof fields state the six
      displayed equations for those very fields.
  \end{itemize}
  In both shapes, the data fields and the equation fields are mandatory. For
  example, if
  $R:\operatorname{LogParameterRegimes}$ is the bundled object, the object must
  contain fields such as
  \[
    R.\operatorname{logn}:\N\to\R,
    \qquad
    R.\operatorname{logn\_eq} :
      \forall n,\ R.\operatorname{logn}(n)=\log n
  \]
  and analogous data/equation pairs for the other five definitions. Merely
  storing proofs that separately declared constants somewhere else have the
  desired equations does not satisfy this node; downstream blueprint nodes must
  be able to project the six objects and their formulas from the declaration
  audited here.

  The six required definitions, with domains and defining equations, are:
  \[
    \logn : \N\to\R,\qquad
    \logn(n) = \log n,
  \]
  \[
    \logDim : \R\to\N\to\N,\qquad
    \logDim(c,n) = \left\lceil c\log n\right\rceil,
  \]
  \[
    \polylogDim : \N\to\N\to\N,\qquad
    \polylogDim(k,n) = \left\lceil(\log n)^k\right\rceil,
  \]
  \[
    \msdHardDim : \R\to\N\to\N,\qquad
    \msdHardDim(c,n) =
      \left\lceil n^{\frac{c}{\log\log n}}\right\rceil,
  \]
  \[
    \mildMsdGamma : \N\to\R,\qquad
    \mildMsdGamma(n) := 1+\frac{1}{\log\log n}.
  \]
  Finally,
  \[
    \msdHardGammaAllowed : \R\to\N\to\R\to\mathrm{Prop}
  \]
  is the fixed-polylogarithmic approximation-regime predicate
  \[
    \msdHardGammaAllowed(C_\gamma,n,\gamma)
    \Longleftrightarrow
    C_\gamma>0\ \wedge\ 1 \le \gamma \ \wedge\
    \gamma \le(\log n)^{C_\gamma}.
  \]
  Here $\logn$ is real-valued, the dimension functions are natural-valued using
  the fixed ceiling convention, and $\msdHardGammaAllowed$ is a proposition over
  a fixed exponent $C_\gamma$ and the pair $(n,\gamma)$. The exponent
  $C_\gamma$ is part of the regime data and is fixed independently of $n$; the
  predicate is not the vacuous pointwise assertion that some exponent can be
  chosen separately for each input size. This is the main-text
  $\gamma\le\operatorname{polylog}(n)$ allowance. The hard dimension
  $n^{c/\log\log n}$ is asymptotically the same as
  $(\log n)^{c\log n/(\log\log n)^2}$ after renaming constants, but the
  declaration audited here must expose the $n^{c/\log\log n}$ formula. It must
  not substitute unrelated expressions such as $c(\log n)^2$.
  Changing ceilings to floors plus a positive lower bound for large $n$ would
  be an equivalent asymptotic convention, but the formal node must still provide
  replacement definitions with the same inputs and outputs.
  The audited declaration must therefore expose all of the following
  field/projection equations, either definitionally or as transparent theorems
  attached to the same declaration:
  \[
  \begin{array}{ll}
    \operatorname{logn\_eq}:&
      \forall n,\ \logn(n)=\log n,\\
    \operatorname{logDim\_eq}:&
      \forall c,n,\ \logDim(c,n)=\lceil c\log n\rceil,\\
    \operatorname{polylogDim\_eq}:&
      \forall k,n,\ \polylogDim(k,n)=\lceil(\log n)^k\rceil,\\
    \operatorname{msdHardDim\_eq}:&
      \forall c,n,\
      \msdHardDim(c,n)=
      \left\lceil n^{\frac{c}{\log\log n}}\right\rceil,\\
    \operatorname{mildMsdGamma\_eq}:&
      \forall n,\ \mildMsdGamma(n)=1+\frac{1}{\log\log n},\\
    \operatorname{msdHardGammaAllowed\_iff}:&
      \forall C_\gamma,n,\gamma,\ 
      \msdHardGammaAllowed(C_\gamma,n,\gamma)\Longleftrightarrow
      C_\gamma>0\wedge 1\le\gamma\wedge
      \gamma\le(\log n)^{C_\gamma}.
  \end{array}
  \]
  A declaration at this label is rejected if it has type `Unit', if it ends in
  `True', if it only states a property of $\msdHardGammaAllowed$, if it only
  packages externally existing constants into a value such as
  `def_log_parameter_regimes : LogParameterRegimes' whose type lacks six data
  projections, if it is a proof-only structure with fields such as
  `logn_eq : logn n = log n' about root identifiers but no corresponding
  `logn' data field, or if its bundled type has the six function fields but
  lacks the six displayed equation fields. The six functions/predicate and their
  formulas must be recoverable from the label declaration itself.

  Computability note. The functions $\logn$, $\logDim$, $\polylogDim$,
  $\msdHardDim$ and $\mildMsdGamma$ are built from $\log$, real division, real
  powers ($\operatorname{rpow}$) and the ceiling $\lceil\cdot\rceil$, all of
  which depend on $\log$, the real $\operatorname{DivInvMonoid}$ instance, or the
  real linear order (used by $\operatorname{Nat.ceil}$, $\max$ and $\min$); these
  are `noncomputable' in Mathlib. Therefore the Lean realizations of these five
  real/natural-valued functions, and of any bundled package value that sets its
  data fields to them (e.g. `def_log_parameter_regimes'), must be declared
  `noncomputable def'. The predicate $\msdHardGammaAllowed$ is a $\mathrm{Prop}$
  and needs no modifier. This annotation does not change any defining equation or
  regime.
\end{definition}

\begin{definition}[Softmax]
  \label{def:softmax}
  This node defines two concrete functions, not just the first one:
  \[
    \operatorname{vectorSoftmax}_n : \R^n\to\R^n,
    \qquad
    \operatorname{matrixSoftmax}_n : \R^{n\times n}\to\R^{n\times n}.
  \]
  For a vector $v \in \R^{n}$,
  \[
    \operatorname{vectorSoftmax}_n(v)
    =
    \frac{(\exp(v[1]), \ldots, \exp(v[n]))}{\sum_{i=1}^{n} \exp(v[i])}
    \in \R^{n}.
  \]
  For a matrix $A \in \R^{n \times n}$ the matrix softmax operator is applied
  row-wise:
  \[
    \operatorname{matrixSoftmax}_n(A)_{i,j}
    =
    \operatorname{vectorSoftmax}_n(A_{i,:})_j
    =
    \frac{\exp(A_{i,j})}{\sum_{k=1}^{n}\exp(A_{i,k})}.
  \]
  We write $\softmax(v)$ and $\softmax(A)$ for these two functions when the
  argument type disambiguates them. A formal declaration at this label must
  expose both the vector function and the row-wise matrix function, together
  with the displayed row equation. A declaration containing only vector softmax
  omits part of Definition~2.1's mathematical input and is not this node.

  Computability note. Both softmax functions use $\exp$ and real division, and
  therefore depend on the real exponential and on the real
  $\operatorname{DivInvMonoid}$ instance, which are `noncomputable' in Mathlib.
  Consequently the Lean realizations of $\operatorname{vectorSoftmax}$,
  $\operatorname{matrixSoftmax}$, and any packaged value bundling them (e.g.
  `def_softmax') must be declared with the `noncomputable' modifier
  (`noncomputable def'). This is a computability annotation only; it does not
  weaken or change the stated equations, and structures merely storing these as
  function-typed fields do not themselves need the modifier.
\end{definition}

\begin{definition}[Attention, Definition 2.1]
  \label{def:attention}
  \uses{def:softmax}
  This node defines the full attention map, not just a container for its
  weights. The declaration audited at this label must be a structured package
  such as `AttentionData' (or an equivalent dependent Sigma package), not only a
  record of positive dimensions and matrices. For positive natural numbers
  $n,d_{\mathrm{in}},d_{\mathrm{out}},m$, an element of the package contains
  all of the following fields:
  \[
  \begin{array}{ll}
    Q,K &: \R^{d_{\mathrm{in}}\times m},\\
    V &: \R^{d_{\mathrm{in}}\times d_{\mathrm{out}}},\\
    M_{Q,K} &: \R^{d_{\mathrm{in}}\times d_{\mathrm{in}}},\\
    S_{Q,K} &: \R^{n\times d_{\mathrm{in}}}\to\R^{n\times n},\\
    A_{Q,K,V} &: \R^{n\times d_{\mathrm{in}}}\to
      \R^{n\times d_{\mathrm{out}}}.
  \end{array}
  \]
  In addition to these object fields, the same audited declaration contains
  named equation fields, for example
  \[
  \begin{array}{ll}
    \operatorname{qkMatrix\_entry}:&
      \forall a,b,\
      (M_{Q,K})_{a,b}=\sum_{r=1}^m Q_{a,r}K_{b,r},\\
    \operatorname{score\_entry}:&
      \forall X,i,j,\
      S_{Q,K}(X)_{i,j}=
      \sum_{a=1}^{d_{\mathrm{in}}}\sum_{b=1}^{d_{\mathrm{in}}}
        X_{i,a}(M_{Q,K})_{a,b}X_{j,b},\\
    \operatorname{apply\_eq}:&
      \forall X,\ A_{Q,K,V}(X)=\softmax(S_{Q,K}(X))XV,\\
    \operatorname{apply\_entry}:&
      \forall X,i,o,\
      A_{Q,K,V}(X)_{i,o}=
      \sum_{j=1}^{n}
        \frac{\exp(S_{Q,K}(X)_{i,j})}
             {\sum_{k=1}^{n}\exp(S_{Q,K}(X)_{i,k})}
        \left(\sum_{a=1}^{d_{\mathrm{in}}}X_{j,a}V_{a,o}\right).
  \end{array}
  \]
  These are fields or transparent projection theorems of the declaration at
  label \ref{def:attention}; they are not optional lemmas to be supplied later.
  Equivalently, the object fields may be transparent definitions rather than
  stored data, but after unfolding the audited declaration the projections
  `qkMatrix', `score' and `apply' must reduce to the formulas above and the four
  named equations must be available at this same node.

  The query-key bilinear matrix is therefore
  \[
    (M_{Q,K})_{a,b}=\sum_{r=1}^m Q_{a,r}K_{b,r}.
  \]
  The score matrix is the function
  $S_{Q,K}(X) \in \R^{n\times n}$ with entry equation
  \[
    S_{Q,K}(X)_{i,j}
    =
    \sum_{a=1}^{d_{\mathrm{in}}}\sum_{b=1}^{d_{\mathrm{in}}}
      X_{i,a}(M_{Q,K})_{a,b}X_{j,b}.
  \]
  The attention map itself is definitionally the function
  \[
    A_{Q,K,V}(X) = \softmax(S_{Q,K}(X)) X V.
  \]
  Equivalently, its entries satisfy the concrete row-wise softmax formula
  \[
    A_{Q,K,V}(X)_{i,o}
    =
    \sum_{j=1}^{n}
      \frac{\exp(S_{Q,K}(X)_{i,j})}
           {\sum_{k=1}^{n}\exp(S_{Q,K}(X)_{i,k})}
      \left(\sum_{a=1}^{d_{\mathrm{in}}} X_{j,a}V_{a,o}\right).
  \]
  This is exactly the usual matrix formula for pairwise query-key scores, but
  the score and output equations are part of the definition. A Lean declaration
  at this label that merely stores $Q,K,V$, even with positivity proofs, is not
  this node. A declaration that stores $Q,K,V$ plus an arbitrary uninterpreted
  run function also omits the mathematical content of Definition~2.1. If the
  formalization uses a structure such as `Attention', that structure must have
  transparent projections or simp theorems for `qkMatrix', `score' and `apply'
  reducing to the displayed formulas above. The audit should reject a structure
  whose only non-proof fields are the matrices $Q,K,V$.
  We write $\mathcal A_{d_{\mathrm{in}}, m, d_{\mathrm{out}}}$ for the set of all
  such attentions.
\end{definition}

\begin{definition}[Multi-layer perceptron, Definition 2.2]
  \label{def:mlp}
  A \emph{multi-layer perceptron} (MLP) is represented by the following concrete
  data:
  \begin{itemize}
    \item natural dimensions $a,b\in\N$;
    \item proofs $0<a$ and $0<b$;
    \item a continuous function $\varphi:\R^{a}\to\R^{b}$.
  \end{itemize}
  The positivity proofs are part of the definition: zero-dimensional source or
  target spaces are not allowed in this blueprint node. This models, following
  the universal approximation theorem, any function approximable by a neural
  network. An MLP is applied to a matrix row-wise: for
  $X \in \R^{n \times a}$,
  \[
    \operatorname{rowApply}_{n,a,b,\varphi}(X)_i=\varphi(X_i),
    \qquad
    \operatorname{rowApply}_{n,a,b,\varphi}(X)\in\R^{n\times b}.
  \]
  A formal `MLP' or `def\_mlp' declaration at this label must therefore store or
  quantify the positive-dimension hypotheses $0<a$ and $0<b$, the continuous
  map, and the row-wise application equation. Allowing arbitrary natural
  dimensions, including $a=0$ or $b=0$, is weaker than this definition.
\end{definition}

\begin{definition}[Flattening matrices for a global output MLP]
  \label{def:matrix-flatten}
  For positive natural numbers $n,d$, this node fixes the standard row-major
  linear equivalence together with its coordinate and continuity data. The
  declaration audited at this label must be a package such as
  `MatrixFlattenData n d' (or an equivalent dependent Sigma package) with named
  fields
  \[
  \begin{array}{ll}
    \operatorname{flatten}_{n,d} &: \R^{n\times d}\simeq \R^{nd},\\
    \operatorname{unflatten}_{n,d} &: \R^{nd}\to\R^{n\times d},\\
    \operatorname{flatten\_coord} &: 
      \operatorname{flatten}_{n,d}(Y)_{(i-1)d+r}=Y_{i,r},\\
    \operatorname{unflatten\_coord} &: 
      \operatorname{unflatten}_{n,d}(z)_{i,r}=z_{(i-1)d+r},\\
    \operatorname{continuous\_flatten} &: 
      \operatorname{Continuous}(\operatorname{flatten}_{n,d}),\\
    \operatorname{continuous\_unflatten} &: 
      \operatorname{Continuous}(\operatorname{unflatten}_{n,d}).
  \end{array}
  \]
  The equivalence is
  \[
    \operatorname{flatten}_{n,d}:\R^{n\times d}\simeq \R^{nd},
    \qquad
    \operatorname{flatten}_{n,d}(Y)_{(i-1)d+r}=Y_{i,r},
  \]
  with inverse $\operatorname{unflatten}_{n,d}$. The package must also prove the
  inverse coordinate equation
  \[
    \operatorname{unflatten}_{n,d}(z)_{i,r}=z_{(i-1)d+r}
  \]
  and the two continuity fields
  \[
    \operatorname{Continuous}(\operatorname{flatten}_{n,d}),
    \qquad
    \operatorname{Continuous}(\operatorname{unflatten}_{n,d}).
  \]
  These continuity proofs are part of the statement at
  label \ref{def:matrix-flatten}. A declaration giving only an equivalence and
  coordinate equations, while proving continuity nowhere or only in later
  helper lemmas not projected from this node, is incomplete.
  This node exists only to make precise how the final MLP of
  Definition~\ref{def:transformer}, whose input is the whole attention-output
  matrix, may be represented as an ordinary vector-input MLP. Any formalization
  that encodes the output MLP as a map $\R^{nd}\to\R$ must include this
  flattening equivalence in the transformer evaluation equation.
\end{definition}

\begin{definition}[Transformer, Definition 2.3]
  \label{def:transformer}
  \uses{def:attention, def:mlp, def:matrix-flatten}
  A \emph{transformer} with one attention unit is the following structured
  object:
  \begin{itemize}
    \item natural dimensions $n,d,d_{\mathrm{in}},m,d_{\mathrm{out}}$ together
      with positivity proofs
      $0<n$, $0<d$, $0<d_{\mathrm{in}}$, $0<m$ and $0<d_{\mathrm{out}}$;
    \item matrices $Q,K\in\R^{d_{\mathrm{in}}\times m}$ and
      $V\in\R^{d_{\mathrm{in}}\times d_{\mathrm{out}}}$ defining the attention
      unit $A_{Q,K,V}$ of Definition~\ref{def:attention};
    \item an input MLP $\varphi_1:\R^d\to\R^{d_{\mathrm{in}}}$ whose stored
      positive dimensions are exactly $d$ and $d_{\mathrm{in}}$, applied
      row-wise to $E\in\R^{n\times d}$ as in Definition~\ref{def:mlp};
    \item an output MLP
      $\varphi_2:\R^{n d_{\mathrm{out}}}\to\R$ whose stored positive dimensions
      are exactly $n d_{\mathrm{out}}$ and $1$.
  \end{itemize}
  The output MLP is global, not row-wise: it receives the entire
  $n\times d_{\mathrm{out}}$ attention-output matrix after applying
  $\operatorname{flatten}_{n,d_{\mathrm{out}}}$. Its run function is not an
  arbitrary field: on an embedding matrix $E \in \R^{n \times d}$ it is
  definitionally
  \[
    \TF(E)
    =
    \varphi_2\!\left(
      \operatorname{flatten}_{n,d_{\mathrm{out}}}
      \left(A_{Q,K,V}(\operatorname{rowApply}_{\varphi_1}(E))\right)
    \right).
  \]
  Thus any formal `Transformer` structure must store the matrices and MLPs (or
  equivalent data from which they can be recovered), the positive-dimension
  hypotheses above, the row-wise application of the input MLP, and the
  flattening equivalence used before the output MLP. Its evaluation equation
  must be the displayed composition. A structure containing only an arbitrary
  function $\R^{n\times d}\to\R$, or an output MLP on $\R^{n d_{\mathrm{out}}}$
  without the displayed flattening/equivalence to $\R^{n\times d_{\mathrm{out}}}$,
  erases architecture that is part of this node.
\end{definition}

\begin{definition}[A transformer solving a decision problem]
  \label{def:tf-solves}
  \uses{def:transformer, def:problem-family}
  Let $\mathcal P$ be a concrete decision problem family, let
  $p:\operatorname{Param}_{\mathcal P}$ be a fixed parameter value, and let
  \[
    \operatorname{embed}_{\mathcal P,p}:
      \operatorname{Inst}_{\mathcal P}(p)\to\R^{n(p)\times d(p)}
  \]
  be the fixed canonical embedding attached to that concrete family and
  parameter. The embedding is only a map from already-valid instances to
  matrices; it does not contain a separate valid-instance predicate and it does
  not redefine the problem semantics. A transformer $\TF$ \emph{solves}
  $\mathcal P$ at the fixed parameter $p$ iff for every instance $I$ of
  $\mathcal P$ at that parameter,
  \[
    \TF(\operatorname{embed}_{\mathcal P,p}(I)) =
    \begin{cases}
      1 & \text{if }\operatorname{Yes}_{\mathcal P}(p,I),\\
      0 & \text{if }\neg\operatorname{Yes}_{\mathcal P}(p,I).
    \end{cases}
  \]
  For the vector problems in this blueprint, $I=(v_1,\ldots,v_n)$ and
  $\operatorname{embed}_{\mathcal P,p}(I)$ is definitionally the matrix whose
  $i$-th row is $v_i$; for document problems it is the matrix of the corresponding
  bag-of-words vectors. This predicate must compare the transformer's output to
  the problem family's own $\operatorname{Yes}$ relation. The tautology
  $\TF(E)=\TF(E)$ is not a solving predicate, and neither is a predicate whose
  truth is controlled by an auxiliary embedding object rather than by
  $\operatorname{Yes}_{\mathcal P}$.
\end{definition}

\begin{definition}[Single-attention-unit realizability predicate]
  \label{def:attention-solves-predicate}
  \uses{def:attention, def:transformer, def:tf-solves, def:problem-family}
  For a concrete decision problem family $\mathcal P$ and a fixed parameter
  value $p$, together with fixed matrix dimensions
  \[
    n(p),\ d(p),\ d_{\mathrm{in}},\ m,\ d_{\mathrm{out}}
  \]
  specified by the surrounding theorem, the assertion
  $\AttentionSolves{\mathcal P(p)}$ means that there exist only the remaining
  architectural witnesses
  \[
    Q,K,V,\ \varphi_1,\ \varphi_2
  \]
  with those fixed dimensions, such that the structured transformer of
  Definition~\ref{def:transformer}, built from the full attention map of
  Definition~\ref{def:attention}, solves the concrete yes/no predicate of
  $\mathcal P$ at the already-fixed parameter $p$ in the sense of
  Definition~\ref{def:tf-solves}. Thus the dimensions are inputs to the
  predicate, not existential witnesses.

  More explicitly, the audited declaration at this label must have the shape of
  a predicate or witness package with fixed-dimension parameters, for example
  \[
  \begin{array}{l}
    \operatorname{def\_attention\_solves}
      (\mathcal P:\operatorname{ConcreteProblemFamily})
      (p:\operatorname{Param}_{\mathcal P})
      (n,d,d_{\mathrm{in}},m,d_{\mathrm{out}}:\N)\\
    \qquad
      (h_n:0<n)(h_d:0<d)(h_{\mathrm{in}}:0<d_{\mathrm{in}})
      (h_m:0<m)(h_{\mathrm{out}}:0<d_{\mathrm{out}})
      (\operatorname{embed}:
        \operatorname{Inst}_{\mathcal P}(p)\to\R^{n\times d})
      : \mathrm{Prop},
  \end{array}
  \]
  and then quantify only
  \[
    Q,K\in\R^{d_{\mathrm{in}}\times m},\quad
    V\in\R^{d_{\mathrm{in}}\times d_{\mathrm{out}}},
    \quad
    \varphi_1:\R^d\to\R^{d_{\mathrm{in}}},
    \quad
    \varphi_2:\R^{n d_{\mathrm{out}}}\to\R .
  \]
  If the implementation uses a witness structure instead of a `Prop', that
  structure must store exactly these fixed-dimension matrices and MLPs, plus
  continuity proofs for the MLPs and the solving proof below. It must not store
  an arbitrary prebuilt `def_transformer' or `Transformer' value whose own
  fields include independent dimensions $n,d,d_{\mathrm{in}},m,d_{\mathrm{out}}$.

  The parameter $p$, the instance type $\operatorname{Inst}_{\mathcal P}(p)$,
  the yes-predicate $\operatorname{Yes}_{\mathcal P}(p,I)$, and the canonical
  embedding $\operatorname{embed}_{\mathcal P,p}$ are inputs to the predicate,
  not existential witnesses. In particular, the predicate must not existentially
  choose an arbitrary `DecisionEmbedding', an arbitrary valid-instance
  predicate, or a different parameter value. For the named vector decision
  problems the canonical embedding is fixed as the row matrix of the given
  vectors; for $\MSD_{n,\ell,t}$ and $\LSD_{n,\ell,t}$ the fixed parameter is
  exactly $(n,\ell,t)$ and the yes/no relation is exactly the cosine-threshold
  relation from Definitions~\ref{def:msd-dec} and~\ref{def:lsd-dec}.

  A witness therefore consists of matrices and continuous MLPs plus a proof
  \[
    \forall I:\operatorname{Inst}_{\mathcal P}(p),\quad
      \TF_{Q,K,V,\varphi_1,\varphi_2}^{n,d,d_{\mathrm{in}},m,d_{\mathrm{out}}}
        (\operatorname{embed}_{\mathcal P,p}(I))=1
      \Longleftrightarrow
      \operatorname{Yes}_{\mathcal P}(p,I),
  \]
  together with the corresponding output-$0$ statement on no instances if the
  formal output is real-valued rather than Boolean. An arbitrary function, a
  constant function, a proof using only an uninterpreted problem tag, or a proof
  made vacuous by choosing an empty auxiliary valid-instance predicate does not
  witness $\AttentionSolves{\mathcal P(p)}$. A declaration that existentially
  quantifies
  \[
    T:\operatorname{def\_transformer}
  \]
  and relies on `def_tf_solves' to relate $T$'s dimensions to $p$ is also
  rejected: the dimension match must be syntactic in this node, by constructing
  $T$ from the fixed-dimension data displayed above. A convenience theorem may
  package those data into a transformer, but its projection equations must show
  that the transformer's dimensions are exactly the fixed inputs and that its
  evaluation reduces to Definition~\ref{def:transformer}'s displayed
  composition.
\end{definition}

\begin{definition}[Bag-of-words embedding]
  \label{def:bow}
  A document is modeled as a finite sequence of tokens over a token type
  $\Sigma$. For a fixed keyword list
  $K:\{1,\ldots,\ell\}\to\Sigma$, define the occurrence predicate
  \[
    \operatorname{Occurs}(D,K_i)
    \Longleftrightarrow
    \exists r,\ D[r]=K_i .
  \]
  The \emph{bag-of-words embedding} is the concrete function
  \[
    \BOW_K : \operatorname{Document}(\Sigma)\to\{0,1\}^{\ell}
  \]
  defined coordinatewise by
  \[
    \BOW_K(D)[i] =
    \begin{cases}
      1 & \text{if }\operatorname{Occurs}(D,K_i),\\
      0 & \text{otherwise.}
    \end{cases}
  \]
  Equivalently, for every document $D$ and every coordinate $i$,
  \[
    \BOW_K(D)[i]=1
    \Longleftrightarrow
    \operatorname{Occurs}(D,K_i),
    \qquad
    \BOW_K(D)[i]=0
    \Longleftrightarrow
    \neg\operatorname{Occurs}(D,K_i).
  \]
  The formal declaration at this label must therefore provide the function
  $\BOW_K$ and the coordinate iff theorem above. A structure that stores only a
  vocabulary, token data, or an occurrence relation without defining the output
  vector and its $0/1$ coordinate behavior is not the bag-of-words embedding.
  For example, a document containing only "There are ten apples on the apple
  tree", with keywords "apple", "tree", "computer", "ten", has embedding
  $(1,1,0,1)$.
\end{definition}

\begin{definition}[Cosine similarity]
  \label{def:cosine}
  \uses{def:inner-prod, def:l2norm}
  This node defines cosine similarity as a function on the subtype of pairs of
  nonzero vectors, not as an unrestricted quotient on all finite real vectors.
  The declaration audited at this label must expose a package
  $\operatorname{CosineData}$, or an equivalent dependent Sigma package, with
  the following named fields:
  \[
  \begin{array}{ll}
    \operatorname{cosine} &: 
      \{(v,w): v,w\in\R^d,\ v\ne0,\ w\ne0\}\to\R,\\
    \operatorname{cosine\_eq} &: 
      \forall v,w,\ v\ne0\to w\ne0,\quad
      \operatorname{cosine}(v,w)=
      \dfrac{\inner{v}{w}}{\norm{v}\norm{w}},\\
    \operatorname{binary\_range} &: 
      \forall v,w\in\{0,1\}^d,\ v\ne0\to w\ne0,\quad
      0\le \operatorname{cosine}(v,w)\le1,\\
    \operatorname{zero\_iff\_inner\_zero} &: 
      \forall v,w\in\{0,1\}^d,\ v\ne0\to w\ne0,\quad
      \operatorname{cosine}(v,w)=0\Longleftrightarrow
      \inner{v}{w}=0.
  \end{array}
  \]
  The nonzero hypotheses are part of the domain and must be visible from this
  declaration. A formalization may implement the function using a subtype
  domain, or by a function taking explicit proofs $v\ne0$ and $w\ne0$, but the
  audited declaration must include the quotient equation and the binary-vector
  range theorem above. A bare definition
  `fun v w => inner v w / (norm v * norm w)' on all real finite vectors is not
  this node, because it erases the nonzero domain and omits the $[0,1]$ range
  property used by the document-similarity problems.

  For nonzero document embeddings $v, w \in \R^{d}$, their \emph{cosine
  similarity} is
  \[
    \cossim{v}{w} = \frac{\inner{v}{w}}{\norm{v}\norm{w}}.
  \]
  This value lies in $[0,1]$ when $v,w \in \{0,1\}^d$; the value $1$ means
  complete similarity and $0$ means no similarity. (Used instead of the raw inner
  product so that two vectors with similar directions but large magnitude are
  still considered close.)

  Computability note. The cosine function uses the Euclidean norm and real
  division, which depend on the real square root and the real
  $\operatorname{DivInvMonoid}$ instance, both `noncomputable' in Mathlib.
  The Lean realization of the $\operatorname{cosine}$ function, and of any
  package value bundling it (e.g. `CosineData'), must therefore be declared
  `noncomputable def'. This annotation does not alter the quotient equation,
  the $[0,1]$ range property, or the zero-iff-inner-zero property required above.
\end{definition}

\begin{lemma}[Cosine as inner product after Euclidean normalization]
  \label{lem:cosine-normalized-dot}
  \uses{def:cosine, def:inner-prod, def:l2norm, lem:inner-scaled}
  For nonzero vectors $v,w \in \R^\ell$, if
  $\widehat v = v/\norm{v}$ and $\widehat w = w/\norm{w}$, then
  $\inner{\widehat v}{\widehat w} = \cossim{v}{w}$.
\end{lemma}
\begin{proof}
  \uses{def:cosine, def:inner-prod, def:l2norm, lem:inner-scaled}
  Apply Lemma~\ref{lem:inner-scaled} with
  $\alpha=\norm{v}$ and $\beta=\norm{w}$. The hypotheses that $v$ and $w$ are
  nonzero give $\norm{v}\ne0$ and $\norm{w}\ne0$, so
  \[
    \inner{v/\norm{v}}{w/\norm{w}}
    =
    \frac{\inner{v}{w}}{\norm{v}\norm{w}}.
  \]
  The right-hand side is exactly $\cossim{v}{w}$ by
  Definition~\ref{def:cosine}.
\end{proof}

\begin{definition}[Most Similar Document, $\MSD_{n,\ell}$, Definition 2.10]
  \label{def:msd}
  \uses{def:cosine, def:problem-family}
  For parameters $n,\ell \in \N$, an instance of $\MSD_{n,\ell}$ is an
  $n$-tuple $(v_1,\ldots,v_n)$ of nonzero vectors in $\{0,1\}^{\ell}$. A valid
  output is a pair $(i,j)$ with $1 \le i,j \le n$, $i \ne j$, such that
  \[
    \cossim{v_i}{v_j}
    =
    \max_{\substack{1\le p,q\le n\\p\ne q}}\cossim{v_p}{v_q}.
  \]
  Formally, this node defines the input predicate ``all rows are nonzero binary
  vectors'', the output type of distinct index pairs, and the
  $\operatorname{ValidOut}$ relation above. Note $\MSD$ differs from $\MaxIP$
  because of the normalization by $\norm{v_i}\norm{v_j}$. A declaration that is
  only a parameter tag named $\MSD$ omits the search relation and is not aligned
  with this definition.

  The declaration audited at this label is a self-contained data package
  `def_msd' (for example a structure value `MSDProblemData'), not
  `def_parameter_sensitive_placeholder', not a call to that placeholder, and not
  a bare `ConcreteProblemFamily' value whose concrete score and optimum are only
  hidden in external helpers. The package may contain a field
  `family : ConcreteProblemFamily' for use by later algorithmic nodes, but the
  audited declaration itself must also expose the concrete fields and projection
  equations below, under these names or transparent equivalents:
  \[
  \begin{array}{ll}
    \operatorname{MSDParam} &= \{(n,\ell): n,\ell\in\N,\ n\ge2,\ \ell\ge1\},\\

      \{(i,j): i,j:\operatorname{Fin} n,\ i\ne j\},\\

      \cossim{V_i}{V_j},\\
    \operatorname{MSDOpt}(V) &=

    \operatorname{MSDValidOut}(n,\ell,V,(i,j))

  \end{array}
  \]
  In field form, the package must contain at least

  \end{array}
  \]
  together with equations identifying these fields with `MSDParam',

  If the implementation packages these names into a `ConcreteProblemFamily',

\end{definition}

\begin{definition}[$\gamma$-MSD, Definition 2.11]
  \label{def:gamma-msd}
  \uses{def:cosine, def:problem-family, def:msd}
  For parameters $n,\ell\in\N$ and approximation factor $\gamma\ge 1$, an
  instance of $\gamma\text{-}\MSD_{n,\ell}$ is an $n$-tuple of nonzero vectors
  $v_1,\ldots,v_n\in\{0,1\}^{\ell}$. A valid output is a pair
  $i^{*}, j^{*}$ with $i^{*}\ne j^{*}$ such that
  \[
    \frac{1}{\gamma}\cdot
    \max_{\substack{1 \le i,j \le n\\i\ne j}} \cossim{v_i}{v_j}
    \le \cossim{v_{i^{*}}}{v_{j^{*}}}
    \le
    \max_{\substack{1 \le i,j \le n\\i\ne j}} \cossim{v_i}{v_j}.
  \]
  The premise $\gamma\ge1$ and the two displayed inequalities are part of the
  formal output relation. A formalization must encode the approximation factor,
  the maximum over distinct pairs, and the returned pair; it cannot replace the
  problem by an uninterpreted family tag.

  The audited declaration is a self-contained data package `def_gamma_msd', not
  `def_parameter_sensitive_placeholder' and not a wrapper around the exact MSD
  family. It must expose a concrete family with parameter $(n,\ell,\gamma)$ and
  the following equations:
  \[
  \begin{array}{ll}
    \operatorname{GammaMSDParam}
      &= \{(n,\ell,\gamma): n,\ell\in\N,\ \gamma\in\R,\
          n\ge2,\ \ell\ge1\},\\
    \operatorname{GammaMSDInst}(n,\ell,\gamma)
      &= \operatorname{MSDInst}(n,\ell),\\
    \operatorname{GammaMSDOut}(n,\ell,\gamma,V)
      &= \{(i,j): i,j:\operatorname{Fin} n,\ i\ne j\},\\
    \operatorname{GammaMSDScore}(V,i,j)
      &= \cossim{v_i}{v_j},\\
    \operatorname{GammaMSDOpt}(V)
      &= M_V=\max_{i\ne j}\cossim{v_i}{v_j},\\
    \operatorname{GammaMSDValidOut}(n,\ell,\gamma,V,(i,j))
      &\Longleftrightarrow
      \gamma\ge1\ \wedge\ i\ne j\ \wedge\
      \gamma^{-1}M_V\le\cossim{v_i}{v_j}\ \wedge\
      \cossim{v_i}{v_j}\le M_V .
  \end{array}
  \]
  In field form, `def_gamma_msd' contains at least

  `family : ConcreteProblemFamily', plus transparent equations identifying all
  of these projections with the displayed `GammaMSD...' objects and with the
  search fields of `family'. The `validOut' field must mention the parameter
  $\gamma$ directly and must include both inequalities
  $\gamma^{-1}M_V\le\cossim{v_i}{v_j}$ and
  $\cossim{v_i}{v_j}\le M_V$. The theorem/projection
  `GammaMSDValidOut_iff' must be available from the audited declaration itself;
  it may not be supplied only by an unrelated external identifier.

  The factor $\gamma$ is a parameter of the family, not metadata outside the
  problem. A declaration that uses abstract components or proves only exact
  maximality does not satisfy this node; the two approximation inequalities
  above must be the visible `ValidOut' relation. A declaration whose audited

  family wrapper, an exact-MSD relation with $\gamma$ ignored, `Unit', or a
  proof ending in `True' is rejected.
\end{definition}

\chapter{Fine-grained complexity: hypotheses and intermediate problems}

% --- The hardness hypotheses, stated as assumptions/conjectures. ------------

\begin{definition}[$k$-SAT]
  \label{def:ksat}
  \uses{def:parameter-sensitive-placeholder}
  For a parameter $k\in\N$, $k\SAT$ is the satisfiability problem for Boolean
  formulas in conjunctive normal form in which every clause has at most
  $k$ literals; $n$ denotes the number of variables. An instance is such a CNF
  formula, and it is a yes instance iff there exists a Boolean assignment of its
  $n$ variables satisfying every clause. Recalled to state SETH. This is a
  concrete decision problem, not merely the symbol $k\SAT$.
\end{definition}

\begin{assumption}[Strong Exponential Time Hypothesis (SETH), Definition 2.4 / A.1]
  \label{assumption:seth}
  \uses{def:ksat, def:truly-subquadratic}
  A $k\SAT$ algorithm is concrete data: a run function on $k$-CNF formulas, a
  proof that the returned Boolean is true exactly on satisfiable formulas, and a
  worst-case runtime function $T_{\mathcal A}(n)$ over formulas with $n$
  variables. SETH is the proposition
  \[
    \forall\,\eps>0,\ \exists\,k\ge1,\ \neg\exists\,\mathcal A,\ C>0,\ n_0,\
      \forall n\ge n_0,\quad
        T_{\mathcal A}(n) \le C\cdot 2^{(1-\eps)n}.
  \]
  Equivalently, for every exponent saving $\eps>0$ there is a clause width
  $k$ for which no correct algorithm decides concrete $k\SAT$ in time
  $O(2^{(1-\eps)n})$. The quantified lower-bound content, the algorithm
  correctness relation for satisfiability, the runtime function, and the
  exponential bound are all part of this assumption. A statement that merely
  quantifies $\eps,k,n$ and concludes `True' for large $n$ is not SETH.
\end{assumption}

\begin{definition}[Orthogonal Vectors, $\OV_{n,\ell}$, Definition 2.5 / A.2]
  \label{def:ov}
  \uses{def:inner-prod, def:parameter-sensitive-placeholder}
  For parameters $n,\ell\in\N$, an instance of $\OV_{n,\ell}$ is an $n$-tuple
  $v_1, \ldots, v_n \in \{0,1\}^{\ell}$. It is a yes instance iff there exists a
  pair $1\le i,j\le n$ with $i\ne j$ and $\inner{v_i}{v_j} = 0$. Thus the
  formal yes-predicate is
  \[
    \exists\, i\ne j,\quad \sum_{r=1}^{\ell} v_i[r]v_j[r]=0.
  \]
  The input type, binary promise, distinctness condition and orthogonality
  equation must all be encoded.
  The formal parameter record for this family contains not only the numeric
  fields $n$ and $\ell$, but also the positivity proofs
  \[
    \operatorname{hn}:2\le n,\qquad
    \operatorname{hell}:1\le\ell .
  \]
  Every construction producing an $\OV$ parameter must fill these fields. In
  particular, a generated record literal of the form
  `{ n := ..., ell := ... }' is incomplete; it must also provide
  `hn := ...' and `hell := ...' proving the displayed bounds.
\end{definition}

\begin{definition}[Bichromatic Orthogonal Vectors]
  \label{def:bichromatic-ov}
  \uses{def:inner-prod, def:parameter-sensitive-placeholder}
  For parameters $n,\ell\in\N$, an instance of bichromatic $\OV_{n,\ell}$ is a
  pair of $n$-tuples $A = \{a_1, \ldots, a_n\}$ and
  $B = \{b_1, \ldots, b_n\}$ of vectors in $\{0,1\}^{\ell}$. It is a yes
  instance iff there exist $i,j$ with $\inner{a_i}{b_j} = 0$. Unlike the
  monochromatic version, $i$ and $j$ range over different color classes and need
  not be distinct. This concrete two-input yes-predicate is the content of the
  node.
  The formal parameter record for this family contains the fields
  \[
    n:\N,\qquad \ell:\N,\qquad
    \operatorname{hn}:1\le n,\qquad
    \operatorname{hell}:1\le\ell .
  \]
  Therefore any parameter map into the bichromatic family must explicitly
  construct the two proof fields. For example, the monochromatic-to-bichromatic
  map from an $\OV$ parameter $p$ is
  \[
    \operatorname{OVToBichromaticParam}(p)=
    \{\,n:=p.n,\ \ell:=p.\ell,\ 
       \operatorname{hn}:=(1\le2\le p.n)\text{ using }p.\operatorname{hn},\
       \operatorname{hell}:=p.\operatorname{hell}\,\}.
  \]
  A record literal that copies only `n' and `ell' omits part of the concrete
  problem parameter and is rejected.
\end{definition}

\begin{conjecture}[Orthogonal Vectors Conjecture (OVC), Conjecture 2.6 / A.3]
  \label{conjecture:ovc}
  \uses{def:ov, def:no-tsq-alg, def:log-parameter-regimes}
  For any $\eps > 0$ there exists a constant $c > 0$ such that
  $\NoTSQAlg{\OV_{n,\logDim(c,n)}}$; equivalently, the dimension is
  $c\log n$ up to the fixed rounding convention.
\end{conjecture}

\begin{assumption}[SETH implies OVC, Williams (2005)]
  \label{thm:seth-implies-ovc}
  \uses{assumption:seth, conjecture:ovc}
  This is a cited external dependency from Williams (2005), not a proof
  reproduced in the paper. Its mathematical content is the implication
  \[
    \text{SETH}\Longrightarrow \text{OVC}.
  \]
  Consequently all hardness results below can be stated under SETH \emph{or}
  OVC. A formalization of this blueprint may register this node as an axiom or
  imported cited theorem. It should not generate a local proof obligation unless
  it also formalizes the standard split-and-list reduction from $k$-SAT to
  Orthogonal Vectors and the corresponding runtime calculation.
\end{assumption}

\begin{definition}[Minimum Inner Product, $\MinIP_{n,\ell}$, Definition 2.7 / A.5]
  \label{def:minip}
  \uses{def:inner-prod, def:parameter-sensitive-placeholder}
  For parameters $n,\ell\in\N$, an instance of $\MinIP_{n,\ell}$ is an $n$-tuple
  $v_1, \ldots, v_n \in \{0,1\}^{\ell}$. A valid output is a distinct pair
  $(i,j)$ minimizing $\inner{v_i}{v_j}$ over all distinct pairs:

\end{definition}

\begin{definition}[$\gamma$-Min-IP, Definition 2.8 / A.6]
  \label{def:gamma-minip}
  \uses{def:inner-prod, def:parameter-sensitive-placeholder}
  For parameters $n,\ell\in\N$ and $\gamma\ge 1$, an instance of
  $\gamma\text{-}\MinIP_{n,\ell}$ is an $n$-tuple
  $v_1, \ldots, v_n \in \{0,1\}^{\ell}$. A valid output is a distinct pair
  $(i,j)$ satisfying the following relation. Define the score and optimum
  \[
    s_I(p,q)=\inner{v_p}{v_q},
    \qquad
    m_I=\min_{\substack{p,q\\p\ne q}}s_I(p,q).
  \]
  Then
  \[
    \operatorname{GammaMinIPValidOut}(n,\ell,\gamma,I,(i,j))
    \Longleftrightarrow
    i\ne j\ \wedge\ \gamma\ge1\ \wedge\
    m_I\le s_I(i,j)\ \wedge\ s_I(i,j)\le\gamma m_I .
  \]
  The premise $\gamma\ge1$ and this multiplicative approximation relation are
  part of the definition. The output score is allowed to be larger than the
  minimum by a factor of at most $\gamma$; it is not required to dominate all
  scores. A declaration that states
  $\forall p,q,\ s_I(p,q)\le s_I(i,j)$, or
  $s_I(i,j)\le\gamma s_I(p,q)$ for every pair $(p,q)$, is a Max-IP-style
  relation and does not satisfy this node. The audited declaration must expose
  the named score $s_I$, optimum $m_I$, and the displayed iff for
  $\operatorname{GammaMinIPValidOut}$.
\end{definition}

\begin{definition}[$\gamma$-Max-IP, Definition A.10]

  $\MaxIP$ relation, not this approximate relation, even though exact
  maximizers are valid examples. The audited declaration must expose the named
  score $s_I$, optimum $M_I$, and the displayed iff for

\begin{lemma}[Append-coordinate construction for bichromatic OV]
  \label{lem:ov-bichromatic-append-construction}
  \uses{def:bichromatic-ov, def:ov, def:inner-prod}
  Given a bichromatic $\OV_{n,\ell}$ instance
  $A=(a_1,\ldots,a_n)$ and $B=(b_1,\ldots,b_n)$, define
  \[
    a_i'=(a_i,1,0)\in\{0,1\}^{\ell+2},
    \qquad
    b_j'=(b_j,0,1)\in\{0,1\}^{\ell+2},
  \]
  and let $W=(a_1',\ldots,a_n',b_1',\ldots,b_n')$ be the corresponding
  monochromatic $\OV_{2n,\ell+2}$ instance. The construction has the following
  concrete behavior:
  \[
    \inner{a_i'}{a_{i'}'}=\inner{a_i}{a_{i'}}+1\ge1,\qquad
    \inner{b_j'}{b_{j'}'}=\inner{b_j}{b_{j'}}+1\ge1,
  \]
  and
  \[
    \inner{a_i'}{b_j'}=\inner{a_i}{b_j}.
  \]
  Consequently,
  \[
    \operatorname{Yes}_{\mathrm{BOV}}(A,B)
    \Longleftrightarrow
    \operatorname{Yes}_{\OV}(W).
  \]
  The formal witness at this label must include the actual transformed
  monochromatic instance as data, together with proof-valued theorem fields for
  every displayed equation and equivalence. Here ``proof-valued'' is meant
  literally: a field named, for example, `appendLeftRow' has type
  \[
    \forall i,\ \operatorname{appendOVInst}(A,B)_{\operatorname{inl}(i)}
      =(a_i,1,0),
  \]
  and the field value is a proof of that statement. It is not a data field
  `appendLeftRow : Prop' whose value is the proposition above, nor a field
  returning a proposition to be proved elsewhere. A generated package whose
  row equations, inner-product equations, binary promise, auxiliary-predicate
  bridge, or yes-equivalence are stored only as `Prop'-valued fields has not
  asserted the construction correctness of this node.

  In Lean terms, the package must contain a field such as
  \[
    \operatorname{appendOVInst}(A,B):
      \operatorname{OVInst}(2n,\ell+2),
  \]
  where the transformed $\OV$ parameter is the full parameter record
  \[
    \operatorname{BOVAppendOVParam}(p)=
    \{\,n:=p.n+p.n,\ \ell:=p.\ell+2,\
       \operatorname{hn}:=2\le p.n+p.n,\
       \operatorname{hell}:=1\le p.\ell+2\,\}.
  \]
  The proof field $\operatorname{hn}$ is obtained from the bichromatic source
  proof $p.\operatorname{hn}:1\le p.n$ twice, giving
  $1+1\le p.n+p.n$. The proof field $\operatorname{hell}$ is obtained from
  $p.\operatorname{hell}:1\le p.\ell$ and monotonicity of addition on the
  right, giving $1\le p.\ell\le p.\ell+2$. Thus a Lean definition
  \[
    \operatorname{BOVAppendOVParam}(p):\operatorname{OVParam}
  \]
  must use a record literal with all four fields `n', `ell', `hn', and
  `hell'. The literal `{ n := p.n + p.n, ell := p.ell + 2 }' is missing the
  proof fields and is not this construction. This is the corresponding
  instance type exposed by Definition~\ref{def:ov}. Its
  row equations must be proof fields of this same package:
  \[
    W_{\operatorname{inl}(i)}=(a_i,1,0),\qquad
    W_{\operatorname{inr}(j)}=(b_j,0,1),
  \]
  equivalently the first $n$ rows are the $a_i'$ and the last $n$ rows are the
  $b_j'$. The package must also include proof fields that $W$ satisfies the
  concrete $\OV$ input promise at parameter $(2n,\ell+2)$, namely that every
  row is a binary vector of length $\ell+2$. The inner-product identities and
  the final correctness field are likewise theorem fields, with types exactly
  the displayed propositions:
  \[
    \operatorname{BichromaticOVYes}_{n,\ell}(A,B)
    \Longleftrightarrow
    \operatorname{OVYes}_{2n,\ell+2}
      (\operatorname{appendOVInst}(A,B)).
  \]
  More explicitly, the audited package has fields, or transparent equivalents,
  with the following \emph{types}:
  \[
  \begin{array}{ll}
    \operatorname{appendLeftRow}:&
      \forall i,\ W_{\operatorname{inl}(i)}=(a_i,1,0),\\
    \operatorname{appendRightRow}:&
      \forall j,\ W_{\operatorname{inr}(j)}=(b_j,0,1),\\
    \operatorname{appendBinaryPromise}:&
      \forall r,h,\ W_r[h]\in\{0,1\},\\
    \operatorname{appendLeftInner}:&
      \forall i,i',\
      \inner{a_i'}{a_{i'}'}=\inner{a_i}{a_{i'}}+1,\\
    \operatorname{appendRightInner}:&
      \forall j,j',\
      \inner{b_j'}{b_{j'}'}=\inner{b_j}{b_{j'}}+1,\\
    \operatorname{appendCrossInner}:&
      \forall i,j,\
      \inner{a_i'}{b_j'}=\inner{a_i}{b_j},\\
    \operatorname{appendPredicate\_iff\_OVYes}:&
      \operatorname{OVAppendMonochromaticYes}(A,B)
      \Longleftrightarrow
      \operatorname{OVYes}_{2n,\ell+2}(W),\\
    \operatorname{appendYesIff}:&
      \operatorname{BichromaticOVYes}_{n,\ell}(A,B)
      \Longleftrightarrow
      \operatorname{OVYes}_{2n,\ell+2}(W).
  \end{array}
  \]
  Each line is a proof field: after projection it is directly usable as a
  theorem of the indicated equality, promise, or iff. The field type may not be
  `Prop', `True', `Unit', a Sigma containing only the proposition as data, or an
  uninterpreted predicate whose definition is not the displayed formula.
  A custom predicate such as
  $\operatorname{OVAppendMonochromaticYes}(A,B)$ may be used only as a local
  abbreviation if the same audited package also proves the definitional
  equality or iff
  \[
    \operatorname{OVAppendMonochromaticYes}(A,B)
    \Longleftrightarrow
    \operatorname{OVYes}_{2n,\ell+2}
      (\operatorname{appendOVInst}(A,B)).
  \]
  Without this bridge to the concrete $\OV$ family, the node is rejected.
  Merely asserting that some transformed instance exists, proving only a
  Sum-indexed auxiliary yes predicate, or stating that the construction is a
  reduction omits the behavior needed by
  Lemma~\ref{lem:ov-bichromatic-equiv}.
\end{lemma}
\begin{proof}
  \uses{def:bichromatic-ov, def:ov, def:inner-prod}
  Expanding the inner product in the last two coordinates gives
  \[
    \inner{(a_i,1,0)}{(a_{i'},1,0)}
      =\inner{a_i}{a_{i'}}+1,
    \qquad
    \inner{(b_j,0,1)}{(b_{j'},0,1)}
      =\inner{b_j}{b_{j'}}+1,
  \]
  so two vectors from the same color class in the transformed monochromatic
  instance can never be orthogonal. Similarly,
  \[
    \inner{(a_i,1,0)}{(b_j,0,1)}
      =\inner{a_i}{b_j}+1\cdot0+0\cdot1
      =\inner{a_i}{b_j}.
  \]
  Therefore the only orthogonal pairs in $W$ are exactly transformed
  cross-color pairs, and the bichromatic and monochromatic yes predicates are
  equivalent.
\end{proof}

\begin{lemma}[OV is subquadratic equivalent to bichromatic OV, Lemma A.4]

    \Longleftrightarrow
    \operatorname{hasZero}(V)\ \vee\
    \operatorname{Yes}_{\mathrm{BOV}}(V,V),
  \]
  together with the proof that when $\operatorname{hasZero}(V)$ is false, every
  diagonal pair has nonzero inner product:
  $\inner{v_i}{v_i}\ne0$ for all $i$.

  The bi-to-mono subpackage has concrete parameter and instance maps
  \[
    (n,\ell)\mapsto(2n,\ell+2),\qquad
    a_i\mapsto a_i'=(a_i,1,0),\qquad
    b_j\mapsto b_j'=(b_j,0,1),
  \]
  where the parameter map is the same full $\operatorname{BOVAppendOVParam}$
  record from Lemma~\ref{lem:ov-bichromatic-append-construction}, including
  the proof fields
  \[
    \operatorname{hn}:2\le n+n,\qquad
    \operatorname{hell}:1\le\ell+2 .
  \]
  and maps $(A,B)$ to the monochromatic list
  $A'\cup B'=(a_1',\ldots,a_n',b_1',\ldots,b_n')$. Its correctness field is the
  pair of displayed equalities and the resulting yes-predicate equivalence for
  the concrete transformed $\OV$ instance:
  \[
    \inner{a_i'}{a_{i'}'}\ge1,\qquad
    \inner{b_j'}{b_{j'}'}\ge1,\qquad
    \inner{a_i'}{b_j'}=\inner{a_i}{b_j},
  \]
  hence
  \[
    \operatorname{Yes}_{\mathrm{BOV}}(A,B)
    \Longleftrightarrow
    \operatorname{Yes}_{\OV}
      \bigl(\operatorname{appendOVInst}(A,B)\bigr),
  \]
  where $\operatorname{appendOVInst}(A,B)$ is the
  $\operatorname{OVInst}(2n,\ell+2)$ from
  Lemma~\ref{lem:ov-bichromatic-append-construction}. The witness may not state
  correctness only for an auxiliary predicate
  $\operatorname{OVAppendMonochromaticYes}$. If such a predicate is introduced,
  the witness must contain a field
  \[
    \operatorname{appendPredicate\_iff\_OVYes}:
    \operatorname{OVAppendMonochromaticYes}(A,B)
    \Longleftrightarrow
    \operatorname{OVYes}_{2n,\ell+2}
      (\operatorname{appendOVInst}(A,B)),
  \]
  and the exported `biToMonoCorrect' field below must target the right-hand
  concrete $\operatorname{OVYes}$ statement, not the auxiliary predicate.
  The Boolean reconstruction field for this direction is exactly the Boolean
  answer returned by the monochromatic OV algorithm on
  $\operatorname{appendOVInst}(A,B)$.

  Both subpackages include proof-valued runtime-transform fields, not just
  correctness:
  \[
    T_{\mathrm{mono}\leftarrow\mathrm{bi}}(n,\ell)
      \le C\bigl(n\ell+T_{\mathrm{bi}}(n,\ell)\bigr),
    \qquad
    T_{\mathrm{bi}\leftarrow\mathrm{mono}}(n,\ell)
      \le C\,T_{\mathrm{mono}}(2n,\ell+2)+C'n\ell,
  \]
  which preserve a truly subquadratic exponent up to polynomial factors in
  $\ell$. These fields must have the displayed inequalities as their field
  types, so projecting the field gives a proof of the inequality. They may not
  be fields of type `Prop' whose values are the displayed propositions, and
  they may not have type `True', `Unit',
  `Nonempty True', or any other vacuous placeholder, and they may not omit the
  target runtime functions $T_{\mathrm{bi}}$ and $T_{\mathrm{mono}}$. The
  transformed runtimes are defined runtimes of the constructed algorithms, not
  arbitrary fields. Concretely, for constants
  $c_{\mathrm{scan}},c_{\mathrm{build}},c_{\mathrm{test}}>0$,
  \[
    T_{\mathrm{mono}\leftarrow\mathrm{bi}}(n,\ell)
      =
      c_{\mathrm{scan}}\,n\ell+
      T_{\mathrm{bi}}(n,\ell)+
      c_{\mathrm{test}}\ell
  \]
  for the zero-vector scan, one bichromatic oracle call and the final Boolean
  reconstruction, while
  \[
    T_{\mathrm{bi}\leftarrow\mathrm{mono}}(n,\ell)
      =
      c_{\mathrm{build}}\,n(\ell+2)+
      T_{\mathrm{mono}}(2n,\ell+2)
  \]
  for building the appended instance and making one monochromatic oracle call.
  The runtime inequalities above are then derived from these definitions. A
  field assigning either transformed runtime to $0$, or to a value independent
  of the target runtime function, is rejected even if the displayed inequality
  is syntactically provable.
  The bi-to-mono runtime field is split into two required fields:
  \[
  \begin{array}{ll}
    \operatorname{biToMonoRuntimeDef}:&
      T_{\mathrm{bi}\leftarrow\mathrm{mono}}(n,\ell,T_{\mathrm{mono}})
      =
      c_{\mathrm{build}}\,n(\ell+2)+T_{\mathrm{mono}}(2n,\ell+2),\\
    \operatorname{biToMonoRuntimeBound}:&
      T_{\mathrm{bi}\leftarrow\mathrm{mono}}(n,\ell,T_{\mathrm{mono}})
      \le C\,T_{\mathrm{mono}}(2n,\ell+2)+C'n\ell .
  \end{array}
  \]
  The left side of the bound must be the named runtime function from
  `biToMonoRuntimeDef'. The bound is rejected if, after unfolding only the field
  declaration being audited, its conclusion has syntactically identical left and
  right sides, or if the target runtime occurs only on the right side but not in
  the definition of
  $T_{\mathrm{bi}\leftarrow\mathrm{mono}}(n,\ell,T_{\mathrm{mono}})$.

  More explicitly, the audited witness package must contain named fields whose
  contents are at least:
  \[
  \begin{array}{ll}
    \operatorname{monoToBiCorrect}:&
      \operatorname{Yes}_{\OV}(V)\Longleftrightarrow
      \operatorname{hasZero}(V)\vee\operatorname{Yes}_{\mathrm{BOV}}(V,V),\\
    \operatorname{monoToBiRuntime}:&
      \forall T_{\mathrm{bi}},\
      T_{\mathrm{mono}\leftarrow\mathrm{bi}}(n,\ell)
      \le C(n\ell+T_{\mathrm{bi}}(n,\ell)),\\
    \operatorname{biToMonoLeftInner}:&
      \inner{a_i'}{a_{i'}'}=\inner{a_i}{a_{i'}}+1,\\
    \operatorname{biToMonoRightInner}:&
      \inner{b_j'}{b_{j'}'}=\inner{b_j}{b_{j'}}+1,\\
    \operatorname{biToMonoCrossInner}:&
      \inner{a_i'}{b_j'}=\inner{a_i}{b_j},\\
    \operatorname{biToMonoCorrect}:&
      \operatorname{Yes}_{\mathrm{BOV}}(A,B)
      \Longleftrightarrow
      \operatorname{OVYes}_{2n,\ell+2}
        (\operatorname{appendOVInst}(A,B)),\\
    \operatorname{appendPredicate\_iff\_OVYes}:&
      \operatorname{OVAppendMonochromaticYes}(A,B)
      \Longleftrightarrow
      \operatorname{OVYes}_{2n,\ell+2}
        (\operatorname{appendOVInst}(A,B))
      \quad\text{if an auxiliary predicate is introduced},\\
    \operatorname{biToMonoRuntime}:&
      \forall T_{\mathrm{mono}},\
      T_{\mathrm{bi}\leftarrow\mathrm{mono}}(n,\ell)
      =c_{\mathrm{build}}n(\ell+2)+T_{\mathrm{mono}}(2n,\ell+2)
      \ \wedge\
      T_{\mathrm{bi}\leftarrow\mathrm{mono}}(n,\ell)
      \le C\,T_{\mathrm{mono}}(2n,\ell+2)+C'n\ell .
  \end{array}
  \]
  The three bi-to-mono inner-product fields may be imported directly from
  Lemma~\ref{lem:ov-bichromatic-append-construction}, but they must still be
  present in, or projected from, the reduction witness audited at this label. A
  proof object containing only empty-regime reductions, only abstract
  $\operatorname{FGReduction}$ fields, only runtime fields equal to `True', or
  runtime fields set to $0$ or otherwise disconnected from the oracle runtime,
  only correctness for $\operatorname{OVAppendMonochromaticYes}$ rather than
  $\operatorname{OVYes}_{2n,\ell+2}(\operatorname{appendOVInst}(A,B))$, or
  only the implication
  $\TSQAlg{\text{bichromatic }\OV}\leftrightarrow\TSQAlg{\OV}$, is too weak and
  does not satisfy this lemma.
\end{lemma}
\begin{proof}
  \uses{def:ov, def:bichromatic-ov, def:inner-prod, lem:ov-bichromatic-append-construction, def:fg-reduction-predicates, def:tsq-alg-exists}
  ($\Leftarrow$) Suppose $\mathcal A$ solves bichromatic $\OV_{n,\ell}$ in
  $\bigO(n^{2-\eps}\operatorname{poly}(\ell))$. Given an $\OV_{n,\ell}$ instance
  $V = \{v_1, \ldots, v_n\}$, first check (in time $\bigO(n\ell)$) whether any
  $v_i$ is the all-zero vector; if so, this is a yes instance. Otherwise set
  $A = B = V$ and run $\mathcal A$ on $A,B$. Since no vector is zero,
  $\inner{v_i}{v_i} \ne 0$ for all $i$, so a pair $i \ne j$ with
  $\inner{v_i}{v_j} = 0$ exists iff $\mathcal A$ reports an orthogonal pair
  $a \in A$, $b \in B$. Total time $\bigO(n^{2-\eps}\operatorname{poly}(\ell) +
  n\ell) = \bigO(n^{2-\eps}\operatorname{poly}(\ell))$.
  This gives the witness
  $R_{\mathrm{mono}\leftarrow\mathrm{bi}}$ on the nonempty source regime of all
  sufficiently large monochromatic parameters; the instance map is $V\mapsto
  (V,V)$ after the explicit zero-vector branch, the output reconstruction is
  the returned Boolean answer, and the displayed runtime bound supplies the
  algorithm-transform field of Definition~\ref{def:fg-reduction-predicates}.

  ($\Rightarrow$) Suppose $\mathcal A'$ solves $\OV_{n,\ell}$ in
  $\bigO(n^{2-\eps}\operatorname{poly}(\ell))$. Given a bichromatic instance
  $A = \{a_1, \ldots, a_n\}$, $B = \{b_1, \ldots, b_n\}$ with $a_i, b_j \in
  \{0,1\}^{\ell}$, build $a_i' = (a_i, 1, 0) \in \R^{\ell+2}$ and
  $b_j' = (b_j, 0, 1) \in \R^{\ell+2}$, and let $A' = \{a_1', \ldots, a_n'\}$,
  $B' = \{b_1', \ldots, b_n'\}$. Then $\inner{a_i'}{a_j'} \ge 1$ and
  $\inner{b_i'}{b_j'} \ge 1$ for all $i,j$ (the appended coordinates contribute
  $1$), while $\inner{a_i'}{b_j'} = \inner{a_i}{b_j}$. Hence a pair of orthogonal
  vectors in $A' \cup B'$ can only be of the form $a_i', b_j'$, i.e.\ an
  orthogonal bichromatic pair $a_i, b_j$. Running $\mathcal A'$ on $A' \cup B'$
  (of size $2n$ in dimension $\ell+2$) decides the bichromatic instance in time
  $\bigO((2n)^{2-\eps}\operatorname{poly}(\ell+2)) =
  \bigO(n^{2-\eps}\operatorname{poly}(\ell))$.
  This gives the witness
  $R_{\mathrm{bi}\leftarrow\mathrm{mono}}$ on the nonempty source regime of all
  sufficiently large bichromatic parameters; its parameter map is
  $(n,\ell)\mapsto(2n,\ell+2)$, its instance map is the append-coordinate
  construction above, its Boolean reconstruction is the monochromatic answer,
  and its correctness theorem is exactly the equivalence between an orthogonal
  pair in $A'\cup B'$ and an orthogonal bichromatic pair in $A\times B$.
\end{proof}

\begin{lemma}[Min-IP is at least as hard as OV]
  \label{lem:minip-harder-ov}
  \uses{def:minip, def:gamma-minip, def:ov, def:fg-reduction-predicates, def:no-tsq-alg}
  For any $\gamma \ge 1$,
  $\AtLeastHard{\MinIP_{n,\ell}}{\OV_{n,\ell}}$ and
  $\AtLeastHard{\gamma\text{-}\MinIP_{n,\ell}}{\OV_{n,\ell}}$. Consequently,
  assuming OVC, for any $\eps>0$ there is $c>0$ such that
  $\NoTSQAlg{\MinIP_{n,c\log n}}$.

  The declaration audited at this label must be the concrete reduction witness
  package $\operatorname{OVToMinIPWitness}$, not only abstract
  $\operatorname{FGReduction}$ fields or the lower-bound conclusion. It contains
  two subwitnesses, one for exact Min-IP and one for $\gamma$-Min-IP. These
  subwitnesses are named
  \[
    \operatorname{exactReduction}
    \quad\text{and}\quad
    \operatorname{gammaReduction},
  \]
  but despite these names their types are not
  $\operatorname{Nonempty}(\operatorname{FGReduction}(\OV,\MinIP))$ and not
  $\operatorname{FGReduction}$ with hidden internals. They are concrete
  structures whose projections are exactly the maps, reconstruction rule,
  zero-correctness theorems and runtime bounds listed below. A generated Lean
  declaration in which `exactReduction' or `gammaReduction' is an abstract
  reduction witness is rejected.

  For every
  $\OV_{n,\ell}$ instance $V=(v_1,\ldots,v_n)$ both subwitnesses use the
  identity instance map
  \[
    \operatorname{toMinIP}(V)=V
  \]
  and, given an oracle output pair $(i,j)$, reconstruct the Boolean OV answer by
  the concrete test
  \[
    \operatorname{answer}(V,(i,j)) =
    \mathrm{true}
    \Longleftrightarrow
    \inner{v_i}{v_j}=0 .
  \]
  The exact subwitness has correctness fields
  \[
    \operatorname{MinInner}(V)=0
    \Longleftrightarrow
    \operatorname{Yes}_{\OV}(V),
    \qquad
    \operatorname{ValidOut}_{\MinIP}(V,(i,j))
    \wedge \inner{v_i}{v_j}=0
    \Longleftrightarrow
    \operatorname{Yes}_{\OV}(V).
  \]
  Here
  \[
    \operatorname{MinInner}(V)
      :=
      \min_{\substack{p,q\\p\ne q}}\inner{v_p}{v_q}
  \]
  is a required field or transparent local definition of the witness package,
  not an opaque symbol.
  The approximate subwitness has the additional zero-approximation field
  \[
    \operatorname{ValidOut}_{\gamma\text{-}\MinIP}(V,(i,j))
    \wedge \gamma\ge1
    \Longrightarrow
    \bigl(\operatorname{MinInner}(V)=0
      \Longleftrightarrow \inner{v_i}{v_j}=0\bigr),
  \]
  which is proved from
  $0\le\operatorname{MinInner}(V)\le\inner{v_i}{v_j}\le
  \gamma\operatorname{MinInner}(V)$. Both subwitnesses include the runtime
  field
  \[
    T_{\OV\leftarrow\MinIP}(n,\ell)
      \le C\bigl(T_{\MinIP}(n,\ell)+n\ell\bigr)
  \]
  and the analogous bound for $\gamma$-Min-IP,
  \[
    T_{\OV\leftarrow\gamma\MinIP}(n,\ell,\gamma)
      \le C\bigl(T_{\gamma\MinIP}(n,\ell,\gamma)+n\ell\bigr).
  \]
  These runtime fields mention the target algorithm runtime function and the
  $O(n\ell)$ cost of testing the returned pair. They are not tautological
  inequalities such as
  $T_{\MinIP}(n,\ell)+n\ell\le T_{\MinIP}(n,\ell)+n\ell$; their left-hand side
  is the runtime of the constructed OV algorithm, defined by running the oracle
  once and then computing $\inner{v_i}{v_j}$ for the returned pair. A proof
  object containing only
  $\operatorname{Nonempty}(\operatorname{FGReduction})$, an opaque algorithm
  transfer implication, or an OVC lower-bound proposition is too weak.

  To make this runtime accounting auditable, the witness must separate the
  definition of the constructed runtime from the upper-bound theorem. The exact
  subwitness contains named projections
  \[
  \begin{array}{ll}
    \operatorname{exactConstructedRuntime}(T_{\MinIP},n,\ell)
      &: \R_{\ge0},\\
    \operatorname{exactRuntimeDef}:&
      \operatorname{exactConstructedRuntime}(T_{\MinIP},n,\ell)
      =
      c_{\mathrm{call}}\,
        T_{\MinIP}(n,\ell)
      +c_{\mathrm{test}}\ell
      +c_{\mathrm{copy}}n\ell
      +c_{\mathrm{bool}},\\
    \operatorname{exactRuntimeBound}:&
      \operatorname{exactConstructedRuntime}(T_{\MinIP},n,\ell)
      \le
      C\bigl(T_{\MinIP}(n,\ell)+n\ell\bigr).
  \end{array}
  \]
  Here $c_{\mathrm{call}},c_{\mathrm{test}},c_{\mathrm{copy}},c_{\mathrm{bool}}$
  are fixed positive constants belonging to the witness. The first summand is
  the runtime of the target Min-IP oracle on the concrete identity instance
  $\operatorname{toMinIP}(V)$, the second is the cost of computing the one
  returned inner product, the third covers copying or coercing the input into
  the target family, and the last covers the Boolean reconstruction. The
  analogous approximate projections are
  \[
  \begin{array}{ll}
    \operatorname{gammaConstructedRuntime}(T_{\gamma\MinIP},n,\ell,\gamma)
      &: \R_{\ge0},\\
    \operatorname{gammaRuntimeDef}:&
      \operatorname{gammaConstructedRuntime}(T_{\gamma\MinIP},n,\ell,\gamma)
      =
      c'_{\mathrm{call}}\,
        T_{\gamma\MinIP}(n,\ell,\gamma)
      +c'_{\mathrm{test}}\ell
      +c'_{\mathrm{copy}}n\ell
      +c'_{\mathrm{bool}},\\
    \operatorname{gammaRuntimeBound}:&
      \operatorname{gammaConstructedRuntime}(T_{\gamma\MinIP},n,\ell,\gamma)
      \le
      C'\bigl(T_{\gamma\MinIP}(n,\ell,\gamma)+n\ell\bigr).
  \end{array}
  \]
  The bound fields are rejected if their left-hand side is syntactically the
  same expression as their right-hand side, if the left-hand side is
  $T_{\MinIP}(n,\ell)+n\ell$ or
  $T_{\gamma\MinIP}(n,\ell,\gamma)+n\ell$ rather than the named constructed
  runtime, or if the named constructed runtime does not mention the target
  oracle runtime. The generated Lean should expose the two defining equations
  and then prove the two inequalities from positivity of the fixed constants;
  it should not satisfy the node by choosing the constructed runtime to be
  exactly the final big-$O$ upper-bound expression.

  These runtime projections are mandatory fields of the subwitnesses
  themselves. In particular, unfolding
  $\operatorname{exactReduction}$ must expose fields literally named, or
  transparently equivalent to,
  `exactConstructedRuntime', `exactRuntimeDef', and `exactRuntimeBound'; and
  unfolding $\operatorname{gammaReduction}$ must expose
  `gammaConstructedRuntime', `gammaRuntimeDef', and `gammaRuntimeBound'. A
  witness that exposes only fields called `runtimeBound' or
  `gammaRuntimeBound', with no separate constructed-runtime function and no
  defining equation for that function, is not a witness for this lemma even if
  the displayed asymptotic inequality appears elsewhere in the file. The audit
  should reject any implementation in which the constructed runtime can be
  changed independently of the oracle call and returned-pair inner-product test,
  or in which the runtime definition is hidden behind an opaque
  $\operatorname{FGReduction}$ object.

  The projections must be separate fields, not a single bundled conjunction.
  In particular, an implementation where the exact side exposes only
  \[
    \operatorname{exactRuntimeBound}(T_{\MinIP})
      \ \wedge\ 
      \forall n,\ell,\ \operatorname{exactRuntimeDef}(T_{\MinIP},n,\ell)
  \]
  is not aligned with this node: the audited structure must have
  `exactConstructedRuntime' as a function field, `exactRuntimeDef' as a theorem
  whose left-hand side is that function, and `exactRuntimeBound' as a theorem
  whose left-hand side is again that same function. The same rule applies to
  the approximate side: `gammaRuntimeBound' alone is insufficient unless
  `gammaConstructedRuntime' and `gammaRuntimeDef' are also projections of
  $\operatorname{gammaReduction}$. The exact and approximate constructed
  runtimes must both contain the oracle-call cost, returned-pair inner-product
  test cost, input-copy/coercion cost, and Boolean reconstruction cost as
  displayed above.
\end{lemma}
\begin{proof}
  \uses{def:minip, def:gamma-minip, def:ov, conjecture:ovc}
  Given an $\OV_{n,\ell}$ instance, run the (approximate) Min-IP algorithm to
  obtain a pair achieving the minimum inner product (or a $\gamma$-approximation
  of it) and test whether that value is $0$. There is an orthogonal pair iff the
  minimum inner product is $0$; and any multiplicative $\gamma$-approximation of
  $0$ must itself be $0$, so the $\gamma$-approximate version detects it too.
  Hence a truly subquadratic algorithm for ($\gamma$-)Min-IP gives one for OV,
  which is impossible under OVC (Conjecture~\ref{conjecture:ovc}).
\end{proof}

\begin{lemma}[Bichromatic Min-IP reduction, Lemma A.8]
  \label{lem:minip-bichromatic-reduction}
  \uses{def:minip, def:gamma-minip, def:minip-dec, def:bichromatic-minip, def:tsq-alg-exists}
  If $\TSQAlg{\text{bichromatic }\MinIP_{n,\ell}}$ exists with polynomial factors
  in $\ell$, then $\TSQAlg{\MinIP_{n,\ell}}$ exists with polynomial factors in
  $\ell$ (possibly with a different positive exponent). The same implication
  holds for $\gamma\text{-}\MinIP_{n,\ell}$ and for the decision version
  $\MinIP_{n,\ell,t}$.

  The audited declaration at this label must expose the recursive halving
  algorithm-transfer package
  $\operatorname{MinIPBichromaticRecursiveWitness}$. This package is the
  declaration audited at the label; the theorem statement is not merely the
  three implications between $\TSQAlg$ propositions. Its fields are:
  \[
    \operatorname{split}(V)=(V_L,V_R),\qquad
    |V_L|=\lceil |V|/2\rceil,\quad |V_R|=\lfloor |V|/2\rfloor,
  \]
  a call tree whose nodes are subsets $U\subseteq V$ and whose non-leaf node
  with split $(U_L,U_R)$ makes exactly the bichromatic oracle call
  $(U_L,U_R)$, and a reconstruction procedure taking all oracle answers and
  returning the best among the cross-split answers and the two recursively
  reconstructed within-half answers. The exact-search correctness field states
  \[
    \min_{\substack{p,q\in U\\p\ne q}}\inner{p}{q}
    =
    \min\!\left(
      \min_{\substack{p\in U_L\\q\in U_R}}\inner{p}{q},
      \min_{\substack{p,q\in U_L\\p\ne q}}\inner{p}{q},
      \min_{\substack{p,q\in U_R\\p\ne q}}\inner{p}{q}
    \right)
  \]
  at every recursive node, together with the theorem that the reconstructed
  pair realizes the global minimum of the original instance. The
  $\gamma$-approximate field states that if every oracle call returns a
  $\gamma$-approximate minimum for its cross part, then the best reconstructed
  candidate is a $\gamma$-approximate minimum for $V$. The threshold-decision
  field states that
  \[
    \exists p\ne q\in U,\ \inner{p}{q}\le t
  \]
  is equivalent to the disjunction of the corresponding cross, left-recursive,
  and right-recursive predicates.
  The cross problem at a non-leaf node is a real target instance, not an index
  or placeholder:
  \[
    \operatorname{crossInst}(U)
      :=(U_L,U_R):\operatorname{BMinIPInst}(|U_L|,|U_R|,\ell),
  \]
  with score
  \[
    s_{\mathrm{cross},U}(p,q)=\inner{(U_L)_p}{(U_R)_q}.
  \]
  The required oracle-call equation is
  \[
    \operatorname{callsAtNode}(U)
      =
      \{\,(|U_L|,|U_R|,\ell,\operatorname{crossInst}(U))\,\}
  \]
  for every non-leaf node. A call field that stores only the node $U$, only the
  pair of cardinalities, or an arbitrary index into the recursion tree is not a
  bichromatic Min-IP oracle call.
  This equation is a named projection of the audited package, for example
  \[
    \operatorname{callsAtNode\_eq\_singleton}(U):
    \operatorname{callsAtNode}(U)
      =
      \{\,(|U_L|,|U_R|,\ell,
          \operatorname{crossInst}(U))\,\}.
  \]
  The right-hand side must contain the actual split lists $U_L,U_R$ through
  $\operatorname{crossInst}(U)$ and the child cardinalities $|U_L|,|U_R|$. A
  statement such as
  $\operatorname{callsAtNode\_eq\_singleton}(U):\mathrm{True}$,
  $\mathrm{True}\wedge\operatorname{Nonempty}(\operatorname{callsAtNode}(U))$,
  or an equality to a singleton whose element is only a node identifier is not
  this projection.
  The parameters of this call are the child cardinalities $|U_L|$ and $|U_R|$,
  or a single balanced-size parameter together with explicit padding/truncation
  equations if the formal bichromatic family is restricted to equal-size color
  classes. They are not the original top-level parameters $(n,\ell)$ unless
  $U$ is the root and the two halves have the required size. Thus a definition
  such as
  \[
    \operatorname{callsAtNode}(U)
      =\{(n,\ell,\operatorname{someGenericBMinIPInst}(n,\ell))\}
  \]
  is rejected: it is not the split subinstance at node $U$.

  Write
  \[
  \begin{array}{ll}
    m_{\mathrm{cross}}(U)&=
      \min_{p\in U_L,\ q\in U_R}\inner{p}{q},\\
    m_L(U)&=\min_{p\ne q\in U_L}\inner{p}{q},\\
    m_R(U)&=\min_{p\ne q\in U_R}\inner{p}{q},\\
    m_U&=\min_{p\ne q\in U}\inner{p}{q}.
  \end{array}
  \]
  The exact-node correctness field is the equation
  \[
    m_U=\min(m_{\mathrm{cross}}(U),\min(m_L(U),m_R(U))).
  \]
  This equation must mention the three separately defined optima above. It is
  rejected if it reduces to
  $m_U=\min(m_U,\min(m_U,m_U))$, or to the same expression with
  `MinIPOpt U' repeated on both sides. The reconstruction fields must define
  named left and right subinstances
  $\operatorname{leftInst}(U)=U_L$ and
  $\operatorname{rightInst}(U)=U_R$, with optima
  $m_L(U)$ and $m_R(U)$ computed on those subinstances, and the cross optimum
  $m_{\mathrm{cross}}(U)$ computed on $\operatorname{crossInst}(U)$. A formal
  proof that obtains the displayed equation by rewriting all three optima to
  the same expression is not an instance of this node.
  The field has the named type
  \[
  \begin{array}{l}
    \operatorname{exactNodeCorrectness}(U):\\
    \qquad
    m_U=\min(m_{\mathrm{cross}}(U),\min(m_L(U),m_R(U))),
  \end{array}
  \]
  where $m_{\mathrm{cross}}(U)$, $m_L(U)$ and $m_R(U)$ are separate projections
  with the displayed definitions. The field is rejected if its type is
  $\mathrm{True}$, $\mathrm{Unit}$, an existential statement with no equality,
  an implication from a validity relation to itself, or any proposition in
  which $m_{\mathrm{cross}}(U)$, $m_L(U)$ and $m_R(U)$ do not all occur.
  The reconstruction fields must define
  \[
    \operatorname{candidateSet}(U)
      =\{o_{\mathrm{cross}}(U),o_L(U),o_R(U)\}
  \]
  with the cross candidate obtained from the valid oracle output for
  $\operatorname{crossInst}(U)$ and the left/right candidates obtained from
  recursive reconstruction, and
  \[
    \operatorname{reconstructExact}(U,\operatorname{answers})
      \in \arg\min_{o\in\operatorname{candidateSet}(U)} s_U(o).
  \]
  The global-correctness field must then prove
  $s_V(\operatorname{reconstructExact}(V,\operatorname{answers}))=m_V$.
  It is not enough to prove an implication whose premise and conclusion are the
  same `ValidOut' proposition.

\begin{proof}
  \uses{def:minip, def:bichromatic-minip}
  Given a $\MinIP_{n,\ell}$ instance $V = \{v_1, \ldots, v_n\}$, partition $V$
  into two halves $V_1, V_2$ of equal size and run $\mathcal A$ on the
  bichromatic instance $(V_1, V_2)$; this returns the minimal pair with one
  element in each half. Then recurse: split $V_1$ into two equal halves and find
  the minimal pair inside $V_1 \times V_1$, and likewise for $V_2$. The minimal
  pair of the whole instance lies either across the top-level split or inside one
  half, so it is found by some recursive call. The running time satisfies
  \[
    \operatorname{poly}(\ell)\cdot \sum_{i=1}^{\log n}
      \bigO\!\left(\left(\tfrac{n}{2^{i}}\right)^{2-\eps}\right)
    = \bigO\!\left(n^{2-\eps}\operatorname{poly}(\ell)\log n\right)
    \le \bigO\!\left(n^{2-\eps'}\operatorname{poly}(\ell)\right)
  \]
  for some $\eps'>0$ (the geometric sum is dominated by its first term). The same
  argument applies to $\gamma\text{-}\MinIP$ (the optimal pair still appears in
  some recursive call) and to $\MinIP_{n,\ell,t}$ (we examine all pairs across
  some call).
\end{proof}

\begin{lemma}[Complement inner product identity]
  \label{lem:complement-inner}
  \uses{def:inner-prod, def:l2norm}
  For binary vectors $v,u\in\{0,1\}^{\ell}$, write $\bar u\in\{0,1\}^{\ell}$ for
  the coordinatewise complement defined by $\bar u[h]=1-u[h]$. If
  $\norm{v}_1=r$, then
  \[
    \inner{v}{\bar u}=r-\inner{v}{u}.
  \]
  This is a concrete arithmetic identity, not an existence claim or a
  placeholder. The declaration audited at this label must state exactly this
  equality, with the complement given by $\bar u[h]=1-u[h]$ and the constant $r$
  fixed as the Hamming weight $\norm{v}_1$. A statement that merely introduces
  $\bar u$, or whose conclusion is $\mathrm{True}$/an $\inner{v}{\bar u}=\inner{v}{\bar u}$
  tautology, does not satisfy this node.
\end{lemma}
\begin{proof}
  \uses{def:inner-prod, def:l2norm}
  Expand the inner product coordinatewise. For each $h$, since $v,u$ are binary,
  \[
    v[h]\,\bar u[h]=v[h](1-u[h])=v[h]-v[h]u[h].
  \]
  Summing over all $h$ and using $\sum_h v[h]=\norm{v}_1=r$ for a binary vector,
  \[
    \inner{v}{\bar u}=\sum_h v[h]-\sum_h v[h]u[h]
      =\norm{v}_1-\inner{v}{u}=r-\inner{v}{u}.
  \]
\end{proof}

\begin{lemma}[Maximum over complements detects orthogonality]
  \label{lem:complement-max-iff-orthogonal}
  \uses{lem:complement-inner, def:inner-prod}
  Fix $0\le r,s\le\ell$ and, for an $\OV$ instance, let
  $S_r,S_s\subseteq\{0,1\}^{\ell}$ be the classes of its vectors of Hamming
  weight $r$ and $s$. For every $v\in S_r$ and every $u\in S_s$,
  \[
    \inner{v}{\bar u}\le r,
    \qquad\text{with equality iff}\qquad
    \inner{v}{u}=0.
  \]
  Consequently, when $S_r$ and $S_s$ are nonempty,
  \[
    \Bigl(\exists v\in S_r,\ \exists u\in S_s,\ \inner{v}{u}=0\Bigr)
    \Longleftrightarrow
    \max_{v\in S_r,\ u\in S_s}\inner{v}{\bar u}=r .
  \]
  The audited declaration must expose \emph{both} the per-pair inequality
  $\inner{v}{\bar u}\le r$ with its equality condition, \emph{and} the
  maximum-equals-$r$ equivalence. A field whose conclusion is $\mathrm{True}$,
  or a tautology of the shape $P\leftrightarrow P$, does not satisfy this node;
  the returned maximum is compared to the concrete weight $r$.
\end{lemma}
\begin{proof}
  \uses{lem:complement-inner, def:inner-prod}
  By Lemma~\ref{lem:complement-inner}, $\inner{v}{\bar u}=r-\inner{v}{u}$. Since
  $v,u$ are binary, $\inner{v}{u}\ge0$, hence $\inner{v}{\bar u}\le r$ with
  equality iff $\inner{v}{u}=0$. Taking the maximum over the nonempty finite set
  $S_r\times S_s$: the maximum equals $r$ iff some pair attains
  $\inner{v}{\bar u}=r$, i.e.\ iff some pair has $\inner{v}{u}=0$.
\end{proof}

\begin{lemma}[Weight-class threshold equivalence for OV]
  \label{lem:ov-weight-class-threshold-equiv}
  \uses{def:ov, def:inner-prod, lem:complement-max-iff-orthogonal}
  Given an $\OV_{n,\ell}$ instance $V=(v_1,\ldots,v_n)$, for $0\le r\le\ell$ let
  $S_r=\{v_i:\norm{v_i}_1=r\}$. Then
  \[
    \operatorname{Yes}_{\OV}(V)
    \Longleftrightarrow
    \exists\,r,s\in\{0,\ldots,\ell\},\
      \max_{v\in S_r,\ u\in S_s}\inner{v}{\bar u}=r .
  \]
  This is the concrete threshold equivalence consumed by
  Lemma~\ref{lem:maxip-harder-ov} as its \texttt{thresholdEquivalence} field.
  The formal declaration audited here must have exactly this type: its left-hand
  side is the $\OV$ yes-predicate of Definition~\ref{def:ov} and its right-hand
  side is the displayed disjunction over weight-class pairs of the
  maximum-equals-weight condition. It must \emph{not} be formalized as the
  tautology
  $\operatorname{Yes}_{\OV}(V)\leftrightarrow\operatorname{Yes}_{\OV}(V)$,
  proved by \texttt{Iff.rfl}, or with $\mathrm{True}$ on either side.
\end{lemma}
\begin{proof}
  \uses{lem:complement-max-iff-orthogonal, def:ov, def:inner-prod}
  Each vector $v_i$ lies in exactly one weight class $S_{\norm{v_i}_1}$, so a
  distinct pair $i\ne j$ with $\inner{v_i}{v_j}=0$ exists iff there are weights
  $r,s$ and vectors $v\in S_r$, $u\in S_s$ with $\inner{v}{u}=0$. By
  Lemma~\ref{lem:complement-max-iff-orthogonal}, applied to the nonempty classes
  (an empty class contributes no such pair and no maximum equal to $r$), this
  holds iff for some $r,s$ the maximum
  $\max_{v\in S_r,\ u\in S_s}\inner{v}{\bar u}$ equals $r$. Hence
  $\operatorname{Yes}_{\OV}(V)$ is equivalent to the displayed disjunction.
\end{proof}

\begin{lemma}[Max-IP is at least as hard as OV, Lemma A.13]

  $\operatorname{OVToBichromaticMaxIPWitness}$. For an OV instance

  \[
    \operatorname{reconstruct}(V,\operatorname{answers})=\mathrm{true}
    \Longleftrightarrow
    \operatorname{OVYes}(V),
  \]

  $\exists r,s,\operatorname{callScore}(V,r,s,o_{r,s})=r$. A field named

  named `thresholdEquivalence' whose type is
  `OVYes V ↔ OVYes V', is explicitly not the reduction correctness theorem.

  `weightCallRuntimeDef',
  `weightCallRuntimeBoundByDoubleSum', and
  `weightCallRuntimeCoarseBound'. The first two must mention the double sum
  $\sum_r\sum_s T_{\mathrm{BMaxIP}}(|S_r|,|S_s|,\ell)$; only the third may use
  a monotonicity-derived coarse charge by
  $(\ell+1)^2T_{\mathrm{BMaxIP}}(n,n,\ell)$.

\begin{proof}
  \uses{def:maxip, def:bichromatic-maxip, def:ov, def:inner-prod, lem:complement-inner, lem:complement-max-iff-orthogonal, lem:ov-weight-class-threshold-equiv}
  Given an $\OV_{n,\ell}$ instance $v_1, \ldots, v_n \in \{0,1\}^{\ell}$,
  partition the vectors into subsets $S_1, \ldots, S_{\ell}$ where $S_j$ contains
  all vectors with exactly $j$ ones. For each pair $1 \le i,j \le \ell$ let
  $\bar S_j$ consist of the vectors in $S_j$ with all entries flipped. Then for
  any $v \in S_i$ and $w \in \bar S_j$ (so $w = \bar u$ with $u \in S_j$), we have
  $\inner{v}{w} = i - \inner{v}{u}$, hence $\inner{v}{\bar u} = i$ iff
  $\inner{v}{u} = 0$. Therefore running the bichromatic Max-IP algorithm
  $\mathcal A$ on $(S_i, \bar S_j)$ and testing whether the maximum inner product
  equals $i$ reveals whether there is an orthogonal pair with one vector in $S_i$
  and one in $S_j$. Doing this for all $\ell^2$ pairs decides OV. The total time
  is $\bigO(\ell^{2}\cdot n^{2-\eps}\operatorname{poly}(\ell)) =
  \bigO(n^{2-\eps}\operatorname{poly}(\ell))$.
\end{proof}

\begin{theorem}[Hardness of $\gamma$-LSD, Theorem 3.1]
  \label{thm:gamma-lsd-hardness}
  \uses{def:gamma-lsd, def:ov, def:cosine, def:inner-prod, assumption:seth, conjecture:ovc, def:no-tsq-alg, def:log-parameter-regimes}
  Assuming SETH or OVC, for every $\eps>0$ there exists a constant $c>0$ such
  that $\NoTSQAlg{\gamma\text{-}\LSD_{n,\ell}}$ for every $\gamma \ge 1$ when
  $\ell = \logDim(c,n)$.

  The statement audited at this label is exactly the concrete lower-bound
  proposition displayed above: for every $\eps>0$ there exists $c>0$ such that
  for every $\gamma\ge1$, $\NoTSQAlg{\gamma\text{-}\LSD_{n,\logDim(c,n)}}$. This
  is a hardness theorem, so its \emph{statement} is the lower-bound proposition
  itself; the reduction package $\operatorname{OVToGammaLSDHardnessWitness}$
  described below is the content of the \emph{proof}, not a field of the
  statement. It is therefore acceptable, and expected, for the audited
  declaration to be the bare lower-bound proposition
  \[
    \forall \eps>0,\ \exists c>0,\ \forall \gamma\ge1,\quad
    \NoTSQAlg{\gamma\text{-}\LSD_{n,\logDim(c,n)}} ,
  \]
  with the witness invoked inside the proof term (for instance through a
  `tsqTransfer' step). What the statement audit forbids is \emph{weakening the
  statement}: the family must be the concrete $\gamma$-LSD family of
  Definition~\ref{def:gamma-lsd} applied to the instantiated triple
  $(n,\logDim(c,n),\gamma)$, the approximation factor must be genuinely
  quantified over all $\gamma\ge1$ (never specialized to $\gamma=1$ only), and
  the dimension must be pinned to $\ell=\logDim(c,n)$. A statement whose
  conclusion is $\NoTSQAlg{\operatorname{def\_gamma\_lsd.family}}$ with an
  uninstantiated family tag, which drops the $\ell=\logDim(c,n)$ regime, or
  which collapses $\forall\gamma\ge1$ to a single $\gamma$, is rejected.

  The \emph{proof} must in turn build and use the reduction package
  $\operatorname{OVToGammaLSDHardnessWitness}$ with all of its concrete maps,
  correctness proofs and runtime transfer below; a proof that asserts the
  lower bound without constructing the OV$\leftarrow\gamma$-LSD reduction is not
  a valid proof of this node.
  Its source slice is the nonempty logarithmic regime
  $\ell=\logDim(c,n)$ for all sufficiently large $n$, and its target parameter
  is the same $(n,\ell,\gamma)$ with $\gamma\ge1$. For an OV instance
  $V=(v_1,\ldots,v_n)$ it defines
  \[
    \operatorname{hasZero}(V)
      \Longleftrightarrow \exists i,\ v_i=0^\ell,
    \qquad
    \operatorname{toLSD}(V)=
      \begin{cases}
        \text{unused} & \text{if }\operatorname{hasZero}(V),\\
        V & \text{otherwise,}
      \end{cases}
  \]
  so the $\gamma$-LSD oracle is called only on the valid nonzero-vector
  instance $V$ when no zero vector is present. Given an oracle output pair
  $(i^*,j^*)$, the reconstructed OV answer is
  \[
    \mathrm{true}
    \Longleftrightarrow
    \operatorname{hasZero}(V)\ \vee\
    \inner{v_{i^*}}{v_{j^*}}=0 .
  \]
  The witness includes an explicit oracle-call field
  \[
    \operatorname{calls}(V)=
      \begin{cases}
        \varnothing & \text{if }\operatorname{hasZero}(V),\\
        \{\,V\,\} & \text{otherwise, viewed as a valid }
          \gamma\text{-}\LSD_{n,\ell}\text{ instance,}
      \end{cases}
  \]
  and an answer-reconstruction field
  \[
    \operatorname{reconstruct}(V,\operatorname{answers})
      =
      \begin{cases}
        \mathrm{true} & \text{if }\operatorname{hasZero}(V),\\
        \bigl(\inner{v_{i^*}}{v_{j^*}}=0\bigr)
          & \text{for the returned LSD pair }(i^*,j^*)\text{ otherwise.}
      \end{cases}
  \]
  Thus a zero-vector OV instance is answered before the LSD oracle is called,
  and a nonzero instance is sent unchanged to the concrete nonzero-binary input
  relation of Definition~\ref{def:gamma-lsd}.
  In the nonzero branch the witness must also expose the concrete oracle answer
  and its validity proof:
  \[
  \begin{array}{ll}
    \operatorname{lsdOracleAnswer}(V,\mathcal A)
      &: \operatorname{GammaLSDOut}(n,\ell,\gamma,V),\\
    \operatorname{lsdOracleCorrect}(V,\mathcal A)
      &: \operatorname{GammaLSDValidOut}
         (n,\ell,\gamma,V,
          \operatorname{lsdOracleAnswer}(V,\mathcal A)),\\
    \operatorname{returnedPair}(V,\mathcal A)
      &:=(i^*,j^*)=
        \operatorname{lsdOracleAnswer}(V,\mathcal A).
  \end{array}
  \]
  The nonzero reconstruction equation is then
  \[
    \operatorname{reconstruct}(V,\operatorname{answers}_{\mathcal A})
    =
    \bigl(\inner{v_{i^*}}{v_{j^*}}=0\bigr),
  \]
  where $(i^*,j^*)$ is exactly the output of the gamma-LSD oracle on $V$. A
  correctness proof that reasons about an arbitrary pair, or about an erased
  answer with no $\operatorname{GammaLSDValidOut}$ proof, does not transfer OV
  correctness.
  The package has explicit parameter-regime fields:
  \[
  \begin{array}{ll}
    \operatorname{sourceSlice}:&
      \ell=\logDim(c,n)\wedge n\ge n_0,\\
    \operatorname{sourceRegime\_iff}:&
      \operatorname{SrcRegime}(n,\ell)
      \Longleftrightarrow n\ge n_0\wedge \ell=\logDim(c,n),\\
    \operatorname{targetParam}:&
      (n,\ell,\gamma)\text{ for the concrete }
      \gamma\text{-}\LSD\text{ family},\\
    \operatorname{targetParam\_eq}:&
      \operatorname{targetParam}(n,\ell,\gamma)=(n,\logDim(c,n),\gamma)
      \text{ on }\operatorname{sourceSlice}.
  \end{array}
  \]
  The variable $\gamma$ in these fields is universally quantified and is the
  same $\gamma$ appearing in the target family, the oracle call, the validity
  relation, the reconstructed algorithm, and the runtime. The witness is
  rejected if any of these fields are specialized to $\gamma=1$, if
  `targetParam' drops $\gamma$, or if the oracle-correctness theorem has the
  type for $\operatorname{GammaLSDValidOut}(n,\ell,1,V,o)$ instead of
  $\operatorname{GammaLSDValidOut}(n,\ell,\gamma,V,o)$ under the explicit
  hypothesis $\gamma\ge1$.
  The exported lower-bound theorem must be parameterized by this same source
  slice and target parameter. It is rejected if its conclusion is merely
  $\NoTSQAlg{\operatorname{def\_gamma\_lsd.family}}$ with no bound
  $(n,\ell,\gamma)$ parameter or no equation $\ell=\logDim(c,n)$.
  The package therefore contains a named lower-bound projection
  \[
  \begin{array}{l}
    \operatorname{lowerBoundOnLogSlice}:\\
    \qquad
    \forall \eps>0,\ \exists c>0,\ \forall n,\ell,\gamma,\ 
    n\ge n_0\to \ell=\logDim(c,n)\to \gamma\ge1\to\\
    \qquad\qquad
    \NoTSQAlg{\gamma\text{-}\LSD_{n,\ell}} .
  \end{array}
  \]
  The conclusion must mention the instantiated parameter triple
  $(n,\ell,\gamma)$ of the concrete $\gamma$-LSD family. It is not enough to
  prove a lower bound for an uninstantiated family value or a proposition whose
  hypotheses do not include $\ell=\logDim(c,n)$.
  Zero-vector correctness is a concrete field, not an excluded case:
  \[
    \operatorname{zeroCaseCorrect}:
      \operatorname{hasZero}(V)\Longrightarrow
      \operatorname{reconstruct}(V,\varnothing)=\mathrm{true}
      \ \wedge\
      \operatorname{Yes}_{\OV}(V).
  \]
  This field must be paired with the defining equation for the zero branch,
  \[
    \operatorname{reconstruct\_zero}(h):
      \operatorname{reconstruct}(V,\varnothing)=\mathrm{true}
      \quad\text{for }h:\operatorname{hasZero}(V),
  \]
  and with a proof
  \[
    \operatorname{hasZero\_to\_OVYes}(h):
      \operatorname{Yes}_{\OV}(V).
  \]
  The proof of `hasZero_to_OVYes' must exhibit an index $i$ with
  $v_i=0^\ell$ and then a distinct partner index $j$ guaranteed by the source
  regime $n\ge2$, showing $\inner{v_i}{v_j}=0$. It is not enough to invoke
  classical excluded middle for $\operatorname{Yes}_{\OV}(V)$.
  For the nonzero branch the witness must contain
  \[
    \operatorname{nonzeroBranchCorrect}:
      \neg\operatorname{hasZero}(V)\Longrightarrow
      \left(
        \operatorname{reconstruct}(V,\operatorname{answers})=\mathrm{true}
        \Longleftrightarrow
        \operatorname{Yes}_{\OV}(V)
      \right).
  \]
  A field with conclusion
  $\operatorname{hasZero}(V)\to
  (\operatorname{Yes}_{\OV}(V)\vee\neg\operatorname{Yes}_{\OV}(V))$ is just
  excluded middle and is rejected; it does not prove that the constructed
  reduction answers true on the zero-vector branch.
  Likewise, a theorem named `zeroCaseCorrect' whose conclusion does not mention
  both `reconstruct' and `OVYes' is rejected, even if a separate lower-bound
  proposition is present.
  The package contains the zero-cosine equivalence
  \[
    v_i\ne0,\ v_j\ne0
    \Longrightarrow
    \bigl(\cossim{v_i}{v_j}=0
      \Longleftrightarrow \inner{v_i}{v_j}=0\bigr),
  \]
  the minimum-zero equivalence
  \[
    \min_{p\ne q}\cossim{v_p}{v_q}=0
    \Longleftrightarrow
    \operatorname{Yes}_{\OV}(V)
  \]
  under $\neg\operatorname{hasZero}(V)$, and the approximation-zero field
  \[
    \operatorname{ValidOut}_{\gamma\text{-}\LSD}(V,(i^*,j^*))
    \wedge \gamma\ge1
    \Longrightarrow
    \left(
      \min_{p\ne q}\cossim{v_p}{v_q}=0
      \Longleftrightarrow
      \cossim{v_{i^*}}{v_{j^*}}=0
    \right).
  \]
  The last field must unfold the $\gamma$-LSD validity relation:
  if
  \[
    m=\min_{p\ne q}\cossim{v_p}{v_q}
    \quad\text{and}\quad
    m\le c^*\le\gamma m,\qquad c^*=\cossim{v_{i^*}}{v_{j^*}},
  \]
  then, using $m\ge0$ for nonzero binary vectors and $\gamma\ge1$,
  \[
    m=0 \Longleftrightarrow c^*=0.
  \]
  This is the multiplicative-zero argument; it cannot be replaced by a generic
  approximation theorem or by an abstract reduction correctness field.
  The nonzero branch correctness field is explicitly derived from these data:
  \[
  \begin{array}{l}
    \operatorname{oracleOutputZeroCorrect}(V,\mathcal A):\\
    \qquad
    \operatorname{GammaLSDValidOut}
      (n,\ell,\gamma,V,(i^*,j^*))
    \to \gamma\ge1\to \neg\operatorname{hasZero}(V)\to\\
    \qquad
    \left(
      \inner{v_{i^*}}{v_{j^*}}=0
      \Longleftrightarrow
      \operatorname{Yes}_{\OV}(V)
    \right).
  \end{array}
  \]
  Its proof must use, in order, the gamma-LSD validity inequalities for the
  returned oracle pair, nonnegativity of binary cosine values, the
  minimum-zero equivalence, and the zero-cosine/zero-inner-product equivalence.
  A field named $\operatorname{nonzeroBranchCorrect}$ that does not take the
  oracle validity proof as an input, or that does not mention
  $\operatorname{GammaLSDValidOut}$, is not this correctness theorem.
  Its runtime field is
  \[
    T_{\OV\leftarrow\gamma\LSD}(n,\ell)
      \le C\bigl(n\ell+T_{\gamma\LSD}(n,\ell,\gamma)\bigr),
  \]
  and it proves this bound preserves the same positive truly-subquadratic
  exponent on the logarithmic dimension slice. The runtime field's left side is
  the runtime of the constructed OV algorithm with the zero-vector scan, at most
  one LSD oracle call, and the final inner-product test. It is not the
  tautology $n\ell+T_{\gamma\LSD}\le n\ell+T_{\gamma\LSD}$ and it is not a
  stand-alone asymptotic lower-bound proposition. A \emph{proof} that concludes
  the high-level lower-bound proposition without constructing these maps,
  equivalences, reconstructed answer and runtime accounting is not a valid proof
  of this theorem.
  As with Lemma~\ref{lem:minip-harder-ov}, this runtime appears through a named
  definition and a separate bound:
  \[
  \begin{array}{ll}
    \operatorname{gammaLSDReductionRuntime}(T_{\gamma\LSD},n,\ell,\gamma)
      &: \R_{\ge0},\\
    \operatorname{gammaLSDRuntimeDef}:&
      \operatorname{gammaLSDReductionRuntime}(T_{\gamma\LSD},n,\ell,\gamma)
      =
      c_{\mathrm{zeroScan}}n\ell+
      c_{\mathrm{call}}T_{\gamma\LSD}(n,\ell,\gamma)+
      c_{\mathrm{test}}\ell+
      c_{\mathrm{branch}},\\
    \operatorname{gammaLSDRuntimeBound}:&
      \operatorname{gammaLSDReductionRuntime}(T_{\gamma\LSD},n,\ell,\gamma)
      \le C\bigl(n\ell+T_{\gamma\LSD}(n,\ell,\gamma)\bigr).
  \end{array}
  \]
  The bound is rejected if the left-hand side is the expression
  $n\ell+T_{\gamma\LSD}(n,\ell,\gamma)$ rather than the named constructed
  runtime, or if the runtime definition omits either the zero-vector scan or the
  target gamma-LSD oracle call.
  The projection named `runtimeTransfer' must be a package or theorem about
  this named runtime, not the final inequality alone. Its statement must expose
  both
  $\operatorname{gammaLSDReductionRuntime}$ and
  $\operatorname{gammaLSDRuntimeDef}$, and its bound conclusion must have
  $\operatorname{gammaLSDReductionRuntime}
  (T_{\gamma\LSD},n,\ell,\gamma)$ as the left-hand side. A proof term whose
  only runtime content is
  $T_{\gamma\LSD}(n,\ell,\gamma)+n\ell
  \le C(T_{\gamma\LSD}(n,\ell,\gamma)+n\ell)$ is a tautology about the oracle
  runtime, not a constructed OV runtime, and is rejected. The projection named
  `lowerBoundStatement' is also accepted only as a derived theorem whose proof
  invokes the concrete `runtimeTransfer', `oracleOutputZeroCorrect',
  `zeroCaseCorrect', and the source-regime fields above. A standalone
  lower-bound proposition attached to the witness, with no definitional path to
  the constructed OV algorithm and its runtime, does not satisfy this theorem.

  Concretely, the runtime transfer package contains all of the following
  projections:
  \[
  \begin{array}{ll}
    \operatorname{constructedOVRun}:&
      \text{scan for a zero vector; otherwise call the }
      \gamma\text{-LSD oracle and test the returned inner product},\\
    \operatorname{constructedOVCorrect}:&
      \operatorname{constructedOVRun}(V)=\mathrm{true}
      \Longleftrightarrow \operatorname{OVYes}(V),\\
    \operatorname{constructedOVRuntime}:&
      \operatorname{constructedOVRuntime}(T_{\gamma\LSD},n,\ell,\gamma)
      =
      \operatorname{gammaLSDReductionRuntime}
        (T_{\gamma\LSD},n,\ell,\gamma),\\
    \operatorname{runtimeTransfer}:&
      \operatorname{gammaLSDReductionRuntime}
        (T_{\gamma\LSD},n,\ell,\gamma)
      \le C(n\ell+T_{\gamma\LSD}(n,\ell,\gamma)).
  \end{array}
  \]
  Each projection must retain the arbitrary $\gamma\ge1$. A proof of the
  theorem for the single case $\gamma=1$ proves only exact LSD hardness and is
  not this node.

  Trial-8 alignment requirement: `runtimeTransfer' is a record whose first
  field is the constructed algorithm, not a bound on a free expression. It must
  expose
  \[
  \begin{array}{ll}
    \operatorname{ovFromGammaLSDRun}(V):&
      \text{if }\operatorname{hasZero}(V)\text{ then true, otherwise }
      \inner{v_{i^*}}{v_{j^*}}=0
      \text{ for the oracle pair }(i^*,j^*),\\
    \operatorname{ovFromGammaLSDRun\_zero}:&
      \operatorname{hasZero}(V)\to
      \operatorname{ovFromGammaLSDRun}(V)=\mathrm{true},\\
    \operatorname{ovFromGammaLSDRun\_nonzero}:&
      \neg\operatorname{hasZero}(V)\to
      \operatorname{ovFromGammaLSDRun}(V)=
      \left(\inner{v_{i^*}}{v_{j^*}}=0\right),\\
    \operatorname{ovFromGammaLSDCorrect}:&
      \operatorname{ovFromGammaLSDRun}(V)=\mathrm{true}
      \Longleftrightarrow \operatorname{OVYes}(V),\\
    \operatorname{ovFromGammaLSDRuntimeDef}:&
      T_{\mathrm{constructed}}(n,\ell,\gamma)
      =
      c_{\mathrm{zeroScan}}n\ell+
      c_{\mathrm{call}}T_{\gamma\LSD}(n,\ell,\gamma)+
      c_{\mathrm{test}}\ell+c_{\mathrm{branch}},\\
    \operatorname{ovFromGammaLSDRuntimeBound}:&
      T_{\mathrm{constructed}}(n,\ell,\gamma)
      \le C(n\ell+T_{\gamma\LSD}(n,\ell,\gamma)).
  \end{array}
  \]
  The field `lowerBoundStatement' is a theorem derived from this record plus
  OVC/SETH; it is not an independent field that may be postulated without a run,
  correctness proof and constructed runtime. The term
  $T_{\gamma\LSD}(n,\ell,\gamma)+n\ell\le
  C(T_{\gamma\LSD}(n,\ell,\gamma)+n\ell)$ may appear only as an intermediate
  inequality after rewriting `ovFromGammaLSDRuntimeDef'; it is not the type of
  `runtimeTransfer'. The nonzero correctness theorem must use the actual oracle
  output and the proof
  $\operatorname{GammaLSDValidOut}(n,\ell,\gamma,V,(i^*,j^*))$; pairwise
  zero-cosine facts without this validity proof do not prove the reduction.
\end{theorem}
\begin{proof}
  \uses{def:gamma-lsd, def:ov, def:cosine, def:inner-prod, conjecture:ovc, thm:seth-implies-ovc, def:log-parameter-regimes}
  Assume for contradiction that some algorithm $\mathcal A$ solves
  $\gamma\text{-}\LSD_{n, \logDim(c,n)}$ in time $\bigO(n^{2-\eps})$ for some
  $\gamma \ge 1$, $\eps>0$ and \emph{every} constant $c>0$. We show that
  $\OV_{n, \logDim(c,n)}$ can then be solved in $\bigO(n^{2-\eps})$ time for
  every $c$, refuting OVC (and hence SETH, by
  Theorem~\ref{thm:seth-implies-ovc}).

  Given $v_1, \ldots, v_n \in \{0,1\}^{\ell}$ with
  $\ell = \logDim(c,n)$: if any vector is the zero vector (checkable in
  $\bigO(n\ell)$ time) output "yes". Otherwise run $\mathcal A$ to obtain
  $i^{*}, j^{*}$ with
  \[
    \min_{1 \le i,j \le n} \cossim{v_i}{v_j}
    \le \cossim{v_{i^{*}}}{v_{j^{*}}}
    \le \gamma \cdot \min_{1 \le i,j \le n} \cossim{v_i}{v_j}.
  \]
  Since all $\norm{v_i} > 0$, the quantity $\min_{i,j}\cossim{v_i}{v_j}$ equals
  $0$ iff there is an orthogonal pair (cosine similarity $0 \Leftrightarrow$
  inner product $0$). As $\gamma \ge 1$, $\cossim{v_{i^{*}}}{v_{j^{*}}} = 0$ iff
  the true minimum is $0$, i.e.\ iff $\mathcal A$ returns an orthogonal pair.
  Thus checking whether $\inner{v_{i^{*}}}{v_{j^{*}}} = 0$ decides
  $\OV_{n,\ell}$. The total time is $\bigO(n\ell + n^{2-\eps}) =
  \bigO(n^{2-\eps})$, refuting SETH/OVC.
\end{proof}

\begin{theorem}[Hardness of LSD and its variants, Corollary 3.2]
  \label{cor:lsd-hardness}
  \uses{def:lsd, def:gamma-lsd, def:bichromatic-gamma-lsd, def:lsd-dec, conjecture:ovc, assumption:seth, def:no-tsq-alg, def:log-parameter-regimes}
  Assuming SETH or OVC, for every $\eps>0$ there exists a constant $c>0$ such
  that $\NoTSQAlg{\LSD_{n,\ell}}$ when $\ell \ge \logDim(c,n)$. Moreover, the
  same lower bound holds for bichromatic
  $\gamma\text{-}\LSD_{n,\ell}$ for all $\gamma \ge 1$, and for $\LSD_{n,\ell,t}$
  for some $t \in [0,1]$.

  The declaration audited at this label must be the corollary witness package
  $\operatorname{LSDHardnessCorollaryWitness}$ with four non-vacuous fields:
  \begin{itemize}
    \item \emph{exact-search transfer}: an algorithm for exact $\LSD_{n,\ell}$
      yields a $\gamma$-LSD algorithm with $\gamma=1$ by running the exact
      search algorithm and returning the same pair, with the proof that exact
      minimizers satisfy the $\gamma=1$ approximation relation;
    \item \emph{larger-dimension transfer}: for every $\ell'\ge\logDim(c,n)$,
      the embedding $v\mapsto(v,0,\ldots,0)\in\{0,1\}^{\ell'}$ preserves
      nonzeroness and all pairwise cosine similarities, so a solver in dimension
      $\ell'$ gives one on the logarithmic hard slice;
    \item \emph{bichromatic transfer}: a solver for bichromatic
      $\gamma$-LSD on two color classes yields a monochromatic $\gamma$-LSD
      solver by the standard recursive split construction over a single list,
      with the same split-tree, reconstruction and runtime accounting as
      Lemma~\ref{lem:minip-bichromatic-reduction}, but using cross-color cosine
      minima instead of inner-product minima;
    \item \emph{threshold transfer}: if every threshold family
      $\LSD_{n,\ell,t}$ for $t\in[0,1]$ had a truly subquadratic algorithm, then
      exact $\LSD_{n,\ell}$ would be solved by enumerating or binary searching
      the finite set
      \[
        \mathcal T_{\ell}
        =
        \left\{\frac{s}{\sqrt{ab}}:
          0\le s\le\ell,\ 1\le a,b\le\ell\right\}
      \]
      of all possible cosine values of nonzero binary vectors and returning a
      pair at the least feasible threshold. The package must therefore prove the
      existential conclusion
      \[
        \exists t\in[0,1],\quad
        \NoTSQAlg{\LSD_{n,\ell,t}}
      \]
      on the logarithmic slice; it must not state a lower bound for an
      unspecified or all-purpose `def_lsd_dec' family without binding the
      witness threshold $t$.
  \end{itemize}
  The witness package itself must contain the concrete algorithms and maps
  listed below. A record with fields only of the form
  `TSQAlg A -> TSQAlg B' or
  `NoTSQAlg A -> NoTSQAlg B' is not this corollary; such implications may be
  exported as derived theorems only after the maps, output reconstruction and
  runtime fields below have been supplied.
  The four fields have the following required concrete shapes:
  \[
  \begin{array}{ll}
    \operatorname{exactToGammaOne}:&
      \operatorname{ValidOut}_{\LSD}(V,(i,j))
      \Rightarrow
      \operatorname{ValidOut}_{1\text{-}\LSD}(V,(i,j)),\\
    \operatorname{padToLargerDim}:&
      v\mapsto(v,0^{\ell'-\ell})\text{ with }
      \cossim{(v,0)}{(w,0)}=\cossim{v}{w},\\
    \operatorname{largerDimRuntime}:&
      T_{\LSD,\ell}(n)\le C(n\ell'+T_{\LSD,\ell'}(n)),\\
    \operatorname{bichromaticSplitTree}:&
      \text{the recursive halving tree over one list }V,\\
    \operatorname{bichromaticCosineNodeCorrectness}:&
      \min_{p\ne q\in U}\cossim{p}{q}
      =
      \min\!\left(
        \min_{p\in U_L,q\in U_R}\cossim{p}{q},
        \min_{p\ne q\in U_L}\cossim{p}{q},
        \min_{p\ne q\in U_R}\cossim{p}{q}
      \right),\\
    \operatorname{bichromaticGammaCorrectness}:&
      \text{best reconstructed cross/recursive candidate is a }
      \gamma\text{-approximate cosine minimum},\\
    \operatorname{thresholdValues}:&
      \mathcal T_\ell=
      \{s/\sqrt{ab}:0\le s\le\ell,\ 1\le a,b\le\ell\},\\
    \operatorname{thresholdSearchCorrectness}:&
      \text{enumerating }\mathcal T_\ell\text{ recovers the least cosine
      value and a minimizing pair},\\
    \operatorname{existsHardThreshold}:&
      \exists t,\ 0\le t\le1\ \wedge\
      \NoTSQAlg{\LSD_{n,\ell,t}}.
  \end{array}
  \]
  More explicitly, `exactToGammaOne' is a witness constructor, not a bare
  implication. Given an exact LSD oracle output $o=(i,j)$, it returns the same
  pair and proves
  \[
    \operatorname{LSDScore}(V,o)=\operatorname{LSDOpt}(V)
    \Longrightarrow
    \operatorname{GammaLSDValidOut}(n,\ell,1,V,o)
  \]
  by unfolding the $\gamma$-LSD inequalities with $\gamma=1$. Its runtime field
  is
  \[
    T_{1\text{-}\LSD}(n,\ell)
      =
      T_{\LSD}(n,\ell)+c_{\mathrm{return}}
      \le C(T_{\LSD}(n,\ell)+1),
  \]
  so the constructed algorithm's runtime is tied to the exact oracle runtime.
  Therefore an `exact-search transfer' field whose type is only
  `TSQAlg LSD -> TSQAlg GammaLSD' is rejected unless the same field, after
  unfolding, exposes this returned pair, validity proof and runtime equation.
  In particular, the witness contains
  \[
  \begin{array}{ll}
    \operatorname{exactToGammaOneRun}(V) &:=
      \operatorname{run}_{\LSD}(V),\\
    \operatorname{exactToGammaOneCorrect}(V) &:\
      \operatorname{LSDValidOut}(V,
        \operatorname{exactToGammaOneRun}(V))
      \to
      \operatorname{GammaLSDValidOut}(n,\ell,1,V,
        \operatorname{exactToGammaOneRun}(V)),\\
    \operatorname{exactToGammaOneRuntimeDef}:&
      T_{1\text{-}\LSD}(n,\ell)
      =
      T_{\LSD}(n,\ell)+c_{\mathrm{return}}.
  \end{array}
  \]
  These are data of the transfer, not consequences hidden behind a
  $\operatorname{Nonempty}$ implication.

  The padding transfer must define, for every $\ell'\ge\ell$,
  \[
    \operatorname{pad}_{\ell,\ell'}(v)
      =(v,0^{\ell'-\ell})
  \]
  together with the three equations
  \[
    \inner{\operatorname{pad}(v)}{\operatorname{pad}(w)}
      =\inner{v}{w},\qquad
    \norm{\operatorname{pad}(v)}=\norm{v},\qquad
    \cossim{\operatorname{pad}(v)}{\operatorname{pad}(w)}
      =\cossim{v}{w}.
  \]
  It then maps an oracle output pair for the padded instance back to the same
  pair of original indices. A padding field that mentions only monotonicity of
  dimension, or only stores an abstract map without these three equations, does
  not prove the larger-dimension claim.
  The larger-dimension lower-bound conclusion must retain the concrete target
  parameter $\ell'$ and the hypothesis $\ell'\ge\logDim(c,n)$. It is rejected
  if it states a lower bound for an unparameterized family `def_lsd.family' or
  if the padding map is not projected from the witness.
  The field named $\operatorname{largerDimensionTransfer}$ must not have type
  $\mathrm{True}$. It is a package with projections
  \[
  \begin{array}{ll}
    \operatorname{padInst}_{\ell,\ell'}(V)
      &: \LSDInst(n,\ell'),\\
    \operatorname{padInst\_row}:&
      \operatorname{padInst}_{\ell,\ell'}(V)_i
      =(v_i,0^{\ell'-\ell}),\\
    \operatorname{padCosine}:&
      \cossim{\operatorname{pad}(v_i)}{\operatorname{pad}(v_j)}
      =\cossim{v_i}{v_j},\\
    \operatorname{unpadOutput}:&
      \operatorname{LSDOut}(n,\ell',
        \operatorname{padInst}_{\ell,\ell'}(V))
      \to \operatorname{LSDOut}(n,\ell,V),\\
    \operatorname{largerDimensionCorrect}:&
      \operatorname{LSDValidOut}(n,\ell',
        \operatorname{padInst}_{\ell,\ell'}(V),o')
      \to
      \operatorname{LSDValidOut}(n,\ell,V,
        \operatorname{unpadOutput}(o')),\\
    \operatorname{largerDimensionRuntimeDef}:&
      T_{\mathrm{pad}}(n,\ell,\ell')
      =
      c_{\mathrm{pad}}n\ell'
      +T_{\LSD,\ell'}(n)
      +c_{\mathrm{unpad}}.
  \end{array}
  \]
  The lower-bound transfer is derived from these projections. A generated field
  whose conclusion is only $\mathrm{True}$, or only says that larger dimension
  is harder without defining $\operatorname{padInst}_{\ell,\ell'}$, is rejected.

  The bichromatic transfer field is a cosine version of the recursive split
  witness. At every node $U=U_L\cup U_R$ it constructs the concrete
  bichromatic $\gamma$-LSD instance $(U_L,U_R)$, receives a valid cross-color
  output, recursively receives valid candidates from $U_L$ and $U_R$, and
  returns the candidate with smallest cosine score among the three. Its
  correctness theorem must use the decomposition
  \[
    m_U=\min(m_{\mathrm{cross}}(U),\min(m_L(U),m_R(U)))
  \]
  where
  $m_{\mathrm{cross}}(U)=\min_{p\in U_L,q\in U_R}\cossim{p}{q}$,
  $m_L(U)=\min_{p\ne q\in U_L}\cossim{p}{q}$, and
  $m_R(U)=\min_{p\ne q\in U_R}\cossim{p}{q}$. It is rejected if it only states
  an implication
  $\TSQAlg{\text{bichromatic }\gamma\text{-}\LSD}\to
    \TSQAlg{\gamma\text{-}\LSD}$, or if its node correctness repeats
  $\operatorname{LSDOpt}(U)$ on all three branches.
  Its call field must be the singleton concrete bichromatic instance
  \[
    \operatorname{cosineCallsAtNode}(U)
      =
      \{\,(|U_L|,|U_R|,\ell,\gamma,(U_L,U_R))\,\},
  \]
  and its reconstruction must choose the smallest cosine-score candidate among
  the cross, left and right answers. These are not optional helper definitions.
  The field named $\operatorname{bichromaticTransfer}$ is not merely
  $\TSQAlg{\text{bichromatic }\gamma\text{-}\LSD}\to
  \TSQAlg{\gamma\text{-}\LSD}$. It must expose the split-tree projections
  \[
  \begin{array}{ll}
    \operatorname{cosineCrossInst}(U)&=(U_L,U_R),\\
    \operatorname{cosineCallsAtNode\_eq}:&
      \operatorname{cosineCallsAtNode}(U)
      =
      \{\,(|U_L|,|U_R|,\ell,\gamma,
          \operatorname{cosineCrossInst}(U))\,\},\\
    \operatorname{cosineCandidateSet}(U)&=
      \{o_{\mathrm{cross}}(U),o_L(U),o_R(U)\},\\
    \operatorname{cosineReconstructGamma}(U,\operatorname{answers})
      &\in
      \arg\min_{o\in\operatorname{cosineCandidateSet}(U)}
        \cossim{\operatorname{left}(o)}{\operatorname{right}(o)}.
  \end{array}
  \]
  Its correctness field is the displayed minimum decomposition for
  $m_U,m_{\mathrm{cross}}(U),m_L(U),m_R(U)$ followed by the
  $\gamma$-approximation inequalities for the reconstructed candidate. A
  $\operatorname{Nonempty}$ implication, or a split field whose node
  correctness is $\mathrm{True}$, does not satisfy this transfer.

  The threshold transfer must bind a specific threshold family. It defines the
  finite set $\mathcal T_\ell$ above, proves that for every nonzero binary
  pair $(v,w)$ the value $\cossim{v}{w}$ belongs to $\mathcal T_\ell$, and
  defines
  \[
    t_{\min}(V)=\min\{t\in\mathcal T_\ell:
      \operatorname{Yes}_{\LSD_{n,\ell,t}}(V)\}.
  \]
  From threshold oracle answers on these finitely many $t$ values it recovers a
  pair attaining $t_{\min}(V)=\operatorname{LSDOpt}(V)$. The hard-threshold
  conclusion is therefore the dependent statement
  \[
    \operatorname{existsHardThreshold}:
    \exists t:\R,\ 0\le t\le1\ \wedge\
      \NoTSQAlg{(\operatorname{LSDThresholdFamily})(n,\ell,t)}.
  \]
  The family in this last display must be the concrete decision family from
  Definition~\ref{def:lsd-dec} instantiated at the returned threshold $t$. A
  conclusion about an uninstantiated `def_lsd_dec' object, or about a threshold
  chosen outside the witness package, is not this field.

\begin{theorem}[Hardness of polylog-approximate $\gamma$-MSD, Theorem 1.2 from Theorem 3.3 / B.1]
  \label{thm:gamma-msd-hardness}
  \uses{def:gamma-msd, assumption:seth, conjecture:ovc, def:no-tsq-alg, def:log-parameter-regimes}
  Assuming SETH or OVC, for every $\eps>0$ and every fixed
  $C_\gamma>0$ there exists a constant $c>0$ such that
  $\NoTSQAlg{\gamma\text{-}\MSD_{n,\msdHardDim(c,n)}}$ whenever
  $\msdHardGammaAllowed(C_\gamma,n,\gamma)$, i.e.
  \[
    \ell = \msdHardDim(c,n)
      = \left\lceil n^{\frac{c}{\log\log n}}\right\rceil
    \quad\text{and}\quad
    1\le\gamma\le(\log n)^{C_\gamma}.
  \]
\end{theorem}
\begin{proof}
  \uses{lem:msd-tensor-reduction, lem:msd-to-additive-msd, lem:additive-msd-maxip-equiv, thm:additive-bmaxip-to-ov, conjecture:ovc, thm:seth-implies-ovc, def:log-parameter-regimes}
  This follows from a combination of Lemma~\ref{lem:msd-tensor-reduction},
  Lemma~\ref{lem:msd-to-additive-msd},
  Lemma~\ref{lem:additive-msd-maxip-equiv} and
  Theorem~\ref{thm:additive-bmaxip-to-ov}. A truly subquadratic algorithm for
  $\gamma\text{-}\MSD_{n,\msdHardDim(c,n)}$ with
  $\msdHardGammaAllowed(C_\gamma,n,\gamma)$ would, by
  Lemma~\ref{lem:msd-tensor-reduction}, give a truly subquadratic algorithm for
  $\mildMsdGamma(n)\text{-}\MSD_{n,\polylogDim(k,n)}$ for any constant $k>0$; by
  Lemma~\ref{lem:msd-to-additive-msd} (which itself combines
  Theorem~\ref{thm:additive-maxip-karthik} and
  Lemma~\ref{lem:additive-msd-maxip-equiv}) this gives a truly subquadratic
  algorithm for bichromatic
  $\frac{\logn(n)}{\logDim(c,n)}$-Additive-$\MSD_{n,\logDim(c,n)}$;
  and finally Theorem~\ref{thm:additive-bmaxip-to-ov} turns a truly subquadratic
  algorithm for the corresponding bichromatic additive Max-IP into a truly
  subquadratic algorithm for $\OV_{n, \logDim(c',n)}$, refuting SETH and OVC
  (Conjecture~\ref{conjecture:ovc}, Theorem~\ref{thm:seth-implies-ovc}). The
  hardness also transfers to harder problems including $\MSD_{n,\ell}$ and
  bichromatic $\gamma\text{-}\MSD_{n,\ell}$.
\end{proof}

\begin{theorem}[Hardness of MSD and its variants, Corollary 3.4]
  \label{cor:msd-hardness}
  \uses{def:msd, def:gamma-msd, def:msd-dec, assumption:seth, conjecture:ovc, def:no-tsq-alg, def:log-parameter-regimes}
  Assuming SETH or OVC, for every $\eps>0$ there exists a constant $c>0$ such

  bichromatic $\gamma\text{-}\MSD_{n,\ell}$ whenever, for some fixed
  $C_\gamma>0$, $\msdHardGammaAllowed(C_\gamma,n,\gamma)$ holds, and for
  $\MSD_{n,\ell,t}$ for some $t \in [0,1]$.
\end{theorem}
\begin{proof}
  \uses{thm:gamma-msd-hardness, def:msd, def:msd-dec, def:no-tsq-alg, def:log-parameter-regimes}

\end{proof}

\begin{definition}[Bichromatic $\gamma$-Additive-MSD]
  \label{def:additive-msd}
  \uses{def:cosine, def:log-parameter-regimes, def:parameter-sensitive-placeholder}
  For parameters $n,\ell\in\N$ and additive gap $\gamma\ge 0$, an instance of
  bichromatic $\gamma$-Additive-MSD consists of two $n$-tuples
  $A=(a_1,\ldots,a_n)$ and $B=(b_1,\ldots,b_n)$ of nonzero vectors in
  $\{0,1\}^{\ell}$ and a threshold $\alpha \in [0,1]$. Let
  $c_I(i,j)=\cossim{a_i}{b_j}$. This is a promise decision family whose two
  predicates are:
  \[
    \operatorname{Yes}_{\gamma}(I)
    \Longleftrightarrow
    \exists\,i,j,\ c_I(i,j)\ge\alpha,
  \]
  and
  \[
    \operatorname{No}_{\gamma}(I)
    \Longleftrightarrow
    \forall\,i,j,\ c_I(i,j)<\alpha-\gamma.
  \]
  A solver for this promise problem must output true on every yes instance and
  false on every no instance; inputs satisfying neither predicate lie in the
  promise gap and need not be classified. The additive gap $\gamma$ appears
  specifically in the no predicate as the separation $\alpha-\gamma$. Recording
  only the yes predicate, or treating $\gamma$ as unused metadata, omits half of
  the problem.
\end{definition}

\begin{definition}[Bichromatic $\gamma$-Additive-Max-IP]
  \label{def:additive-maxip}
  \uses{def:inner-prod, def:log-parameter-regimes, def:parameter-sensitive-placeholder}
  For parameters $n,\ell\in\N$ and additive gap $\gamma\ge 0$, an instance of
  bichromatic $\gamma$-Additive-Max-IP ($\gamma$-Additive-BMax-IP) consists of
  two $n$-tuples $A=(a_1,\ldots,a_n)$ and $B=(b_1,\ldots,b_n)$ in
  $\{0,1\}^{\ell}$ and a threshold $\alpha$. Let
  $s_I(i,j)=\inner{a_i}{b_j}$. This is a promise decision family whose two
  predicates are:
  \[
    \operatorname{Yes}_{\gamma}(I)
    \Longleftrightarrow
    \exists\,i,j,\ s_I(i,j)\ge\alpha,
  \]
  and
  \[
    \operatorname{No}_{\gamma}(I)
    \Longleftrightarrow
    \forall\,i,j,\ s_I(i,j)<\alpha-\gamma.
  \]
  A solver must output true on all yes instances and false on all no instances;
  inputs in the gap between $\alpha-\gamma$ and $\alpha$ are outside the promise.
  The formal problem must store both vector lists, the threshold $\alpha$, both
  predicates, and the additive gap $\gamma$. A declaration containing only the
  existential yes predicate is not the additive-gap problem used by Lemma B.6.
\end{definition}

\begin{lemma}[Tensoring amplifies the MSD approximation factor, Lemma B.2]
  \label{lem:msd-tensor-reduction}

\begin{proof}
  \uses{lem:kron-power-cosine, def:gamma-msd, def:kron, def:cosine, def:log-parameter-regimes}
  Let $v_1, \ldots, v_n \in \{0,1\}^{\polylogDim(k,n)}$ be an instance of
  $\mildMsdGamma(n)\text{-}\MSD_{n,\polylogDim(k,n)}$. Construct
  $V' = \{v_1^{\otimes q}, \ldots, v_n^{\otimes q}\}$ for
  $q = \frac{\log n}{(\log\log n)^{2}}$. The vectors of $V'$ have dimension
  \[
    (\log n)^{kq}
      =(\log n)^{\frac{k\log n}{(\log\log n)^2}}
      =n^{\frac{k}{\log\log n}},
  \]
  so, up to the fixed ceiling convention and a harmless renaming of the constant
  $c$, this is exactly the hard dimension
  $\msdHardDim(k,n)=\lceil n^{k/\log\log n}\rceil$. Run the assumed hard-regime
  algorithm on $V'$ with any fixed polylogarithmic approximation factor
  $\Gamma(n)=(\log n)^{C_\gamma}$ satisfying
  $\msdHardGammaAllowed(C_\gamma,n,\Gamma(n))$. By
  Lemma~\ref{lem:kron-power-cosine},
  \[
    \frac{\inner{v_i^{\otimes q}}{v_j^{\otimes q}}}
         {\norm{v_i^{\otimes q}}\cdot\norm{v_j^{\otimes q}}}
    = \left(\cossim{v_i}{v_j}\right)^{q}
  \]
  for all $i,j$. Therefore a $\Gamma(n)$-approximation of MSD on $V'$ is a
  $\Gamma(n)^{1/q}$-approximation of MSD on $v_1,\ldots,v_n$. Since
  \[
    \Gamma(n)^{1/q}
      =(\log n)^{C_\gamma(\log\log n)^2/\log n}
      \le 1+\frac{1}{\log\log n}
      =\mildMsdGamma(n)
  \]
  for all sufficiently large $n$, the returned pair is also a valid
  $\mildMsdGamma(n)$-approximation. The running time is preserved up to the
  $\bigO(n^{2-\eps})$ bound.
\end{proof}

\begin{lemma}[Additive-MSD is equivalent to Additive-Max-IP, Lemma B.5]
  \label{lem:additive-msd-maxip-equiv}
  \uses{def:additive-msd, def:additive-maxip, def:fg-reduction-predicates, def:log-parameter-regimes}
  $\SubquadEquiv{\text{bichromatic }\frac{\logn(n)}{\ell}\text{-Additive-}\MSD_{n,\ell}}
  {\text{bichromatic }\logn(n)\text{-Additive-Max-IP}_{n,\ell'}}$ when
  $\ell = \bigO(\log n)$ and $\ell' = \bigO(\log n)$.
\end{lemma}
\begin{proof}
  \uses{def:additive-msd, def:additive-maxip, def:inner-prod, def:l2norm, def:log-parameter-regimes}
  ($\Leftarrow$) Given a bichromatic $\logn(n)$-Additive-Max-IP$_{n,\ell'}$
  instance with sets $A = \{a_1, \ldots, a_n\}$, $B = \{b_1, \ldots, b_n\}$ and
  threshold $\alpha$, construct $A' = \{a_1', \ldots, a_n'\}$, $B' = \{b_1',
  \ldots, b_n'\}$ as follows: for each $a_i$ first attach $\ell - \norm{a_i}_1$
  ones at the end and then another $\ell + \norm{a_i}_1$ zeros to obtain $a_i'
  \in \{0,1\}^{3\ell}$, and for each $b_j$ attach $\ell + \norm{b_j}_1$ zeros and
  then another $\ell - \norm{b_j}_1$ ones to obtain $b_j' \in \{0,1\}^{3\ell}$.
  Now every $a_i'$ and $b_j'$ has exactly $\ell$ ones, so $\norm{a_i'} =
  \norm{b_j'} = \sqrt{\ell}$, and $\inner{a_i'}{b_j'} = \inner{a_i}{b_j}$ by the
  construction (the appended coordinates never overlap on a $1$). Hence running an
  $\frac{\logn(n)}{\ell}$-Additive-MSD algorithm on $A', B'$ with threshold
  $\frac{\alpha}{\ell}$ distinguishes:
  \begin{itemize}
    \item there exists $(a',b')$ with $\frac{\inner{a'}{b'}}{\norm{a'}\norm{b'}}
      \ge \frac{\alpha}{\ell}$, equivalent to $\inner{a}{b} \ge \alpha$; or
    \item for all $(a',b')$, $\frac{\inner{a'}{b'}}{\norm{a'}\norm{b'}} <
      \frac{\alpha}{\ell} - \frac{\logn(n)}{\ell}$, equivalent to
      $\inner{a}{b} < \alpha - \logn(n)$ for all $a \in A$, $b \in B$.
  \end{itemize}
  The running time is $\bigO(n^{2-\eps}\operatorname{poly}(3\ell)) =
  \bigO(n^{2-\eps}\operatorname{poly}(\ell))$.

  ($\Rightarrow$) Conversely, given a bichromatic
  $\frac{\logn(n)}{\ell}$-Additive-MSD$_{n,\ell}$ instance with sets $A, B$ and
  threshold $\alpha$, build the exact same $A', B'$ so that
  $\inner{a_i'}{b_j'} = \inner{a_i}{b_j}$ and $\norm{a_i'} = \norm{b_j'} =
  \sqrt{\ell}$ for all $i,j$. The same argument shows that running an
  Additive-Max-IP algorithm on $A', B'$ with threshold $\alpha\cdot\ell$ solves
  the Additive-MSD instance.
\end{proof}

\begin{theorem}[An attention unit solves OV, Theorem 4.1]
  \label{thm:attention-solves-ov}
  \uses{def:transformer, def:tf-solves, def:ov, def:attention-solves-predicate}
  An attention unit with input and output MLPs and parameters $d = \ell$,
  $d_{\mathrm{in}} = \ell$, $d_{\mathrm{out}} = 1$, $m \ge \ell + 1$ can solve
  $\OV_{n,\ell}$; in the notation of
  Definition~\ref{def:attention-solves-predicate}, $\AttentionSolves{\OV_{n,\ell}}$.
\end{theorem}
\begin{proof}
  \uses{def:attention, def:softmax, def:tf-solves, def:ov, lem:mlp-threshold, def:inner-prod}
  Let $v_1, \ldots, v_n \in \{0,1\}^{\ell}$ be an $\OV_{n,\ell}$ instance; define
  $v_{n+1} := 0^{\ell}$ and let $X \in \R^{(n+1)\times\ell}$ have rows
  $X_{i,:} = v_i$. Since $m \ge \ell + 1$, choose $Q, K \in \R^{\ell \times m}$
  whose score bilinear form satisfies $M_{Q,K} = -3\log n \cdot I_{\ell}$, let
  $V \in \R^{\ell\times 1}$ be the all-one matrix, and take $\varphi_1$ to be the
  identity. We claim that if there is a pair $i \ne j$ with
  $\inner{v_i}{v_j} = 0$ then $A_{Q,K,V}(X)$ has an entry $\ge \frac{1}{n+1}$,
  and otherwise all entries are $\le
  \frac{1}{(n+1)^{1.5}}$; the second MLP $\varphi_2$ then outputs $1$ iff some
  entry is $\ge \frac{1}{n+1}$ (existence guaranteed by
  Lemma~\ref{lem:mlp-threshold}).

  The $i$-th entry of $A_{Q,K,V}(X)$ equals
  \[
    \sum_{j=1}^{n} \frac{\exp(-3\log n \cdot \inner{v_i}{v_j})}
      {\sum_{k=1}^{n}\exp(-3\log n \cdot \inner{v_i}{v_k}) + 1}\cdot \norm{v_j}_1
    = \frac{\sum_{j=1}^{n} n^{-3\inner{v_i}{v_j}}\cdot \norm{v_j}_1}
           {\sum_{k=1}^{n} n^{-3\inner{v_i}{v_k}} + 1},
  \]
  where $\norm{v_j}_1 = (XV)_j$ (the all-one $V$ sums the entries of $v_j$) and
  the $+1$ in the denominator is the contribution of $v_{n+1} = 0^{\ell}$ (giving
  $n^{0} = 1$). If there is a pair with $\inner{v_{i^{*}}}{v_{j^{*}}} = 0$, the
  $i^{*}$-th entry is at least
  \[
    \frac{n^{-3\inner{v_{i^{*}}}{v_{j^{*}}}}\cdot \norm{v_{j^{*}}}_1}{n+1}
    \ge \frac{1}{n+1},
  \]
  since $\norm{v_{j^{*}}}_1 \ge 1$ and the denominator is at most $n+1$.
  Otherwise $\inner{v_i}{v_j} \ge 1$ for all $i,j$, so the $i$-th entry is at most
  \[
    \sum_{j=1}^{n} \frac{n^{-3\inner{v_i}{v_j}} \cdot \norm{v_j}_1}{1}
    \le \sum_{j=1}^{n} \frac{\ell}{n^{3\inner{v_i}{v_j}}}
    \le n\cdot \frac{\ell}{n^{3}} < \frac{1}{n^{1.5}},
  \]
  using $\norm{v_j}_1 \le \ell$ and $\ell = \bigO(\operatorname{polylog} n)$.
\end{proof}

\begin{theorem}[An attention unit solves Max-IP and Min-IP decision, Theorem C.1]
  \label{thm:attention-solves-maxip-minip}
  \uses{def:transformer, def:tf-solves, def:maxip-dec, def:minip-dec, def:attention-solves-predicate}
  An attention unit with input and output MLPs and parameters $d = \ell$,
  $d_{\mathrm{in}} = \ell + 1$, $d_{\mathrm{out}} = 1$, $m \ge \ell + 1$ can
  solve $\MaxIP_{n,\ell,t}$ and $\MinIP_{n,\ell,t}$ for $1 \le t \le \ell$; that
  is, $\AttentionSolves{\MaxIP_{n,\ell,t}}$ and
  $\AttentionSolves{\MinIP_{n,\ell,t}}$.
\end{theorem}
\begin{proof}
  \uses{def:attention, def:softmax, def:maxip-dec, def:minip-dec, lem:mlp-threshold, lem:mlp-append, def:inner-prod}
  \textbf{Max-IP.} Given $v_1, \ldots, v_n \in \{0,1\}^{\ell}$ and $V \in
  \R^{n\times\ell}$ with $V_{i,:} = v_i$, set $x_i = (v_i, 1) \in
  \{0,1\}^{\ell+1}$, $x_{n+1} = (0, \ldots, 0, t+1) \in \R^{\ell+1}$, and let
  $X \in \R^{(n+1)\times(\ell+1)}$ have rows $X_{i,:} = x_i$. Choose $Q, K \in
  \R^{(\ell+1)\times m}$ whose score bilinear form satisfies
  $M_{Q,K} = 3\log n \cdot I_{\ell+1}$, and
  $\mathcal V = (1, \ldots, 1, 0) \in \R^{\ell+1}$. The input MLP $\varphi_1$
  produces the rows $x_i$ from the $v_i$ via Lemma~\ref{lem:mlp-append}. Then the
  $i$-th entry of $A_{Q,K,\mathcal V}(X)$ equals
  \[
    \frac{\sum_{j=1}^{n} n^{3\inner{x_i}{x_j}}\cdot \norm{v_j}_1}
         {\sum_{k=1}^{n} n^{3\inner{x_i}{x_k}} + n^{3(t+1)}}.
  \]
  Let $i^{*} \ne j^{*}$ maximize $\inner{v_{i^{*}}}{v_{j^{*}}}$. If
  $\inner{v_{i^{*}}}{v_{j^{*}}} \ge t$, then $\inner{x_{i^{*}}}{x_{j^{*}}} \ge
  t+1$ and the $i^{*}$-th entry is at least
  \[
    \frac{n^{3\inner{x_{i^{*}}}{x_{j^{*}}}}\cdot \norm{v_{j^{*}}}_1}
         {\sum_{k} n^{3\inner{x_{i^{*}}}{x_k}} + n^{3(t+1)}}
    \ge \frac{\norm{v_{j^{*}}}_1}{n+1} \ge \frac{1}{n+1}.
  \]
  If instead $\inner{v_i}{v_j} \le t-1$ for all $i \ne j$, then
  $\inner{x_i}{x_j} \le t$ and the $i$-th entry is at most
  \[
    \frac{\sum_{j=1}^{n} n^{3\inner{x_i}{x_j}}\cdot \norm{v_j}_1}{n^{3(t+1)}}
    \le \frac{\ell\cdot n\cdot n^{3t}}{n^{3(t+1)}} = \frac{\ell}{n^{2}}
    < \frac{1}{(n+1)^{1.5}}.
  \]
  Thus $\varphi_2$ outputs $1$ iff some entry is $\ge \frac{1}{n+1}$
  (Lemma~\ref{lem:mlp-threshold}), solving $\MaxIP_{n,\ell,t}$.

  \textbf{Min-IP.} The proof is the same except the score bilinear form is
  $M_{Q,K} = -3\log n \cdot I_{\ell+1}$ and
  $x_{n+1} = (0, \ldots, 0, -(t+1)) \in \R^{\ell+1}$. The $i$-th entry of
  $A_{Q,K,\mathcal V}(X)$ becomes
  \[
    \frac{\sum_{j=1}^{n} n^{-3\inner{x_i}{x_j}}\cdot \norm{v_j}_1}
         {\sum_{k=1}^{n} n^{-3\inner{x_i}{x_k}} + n^{-3(t+1)}}.
  \]
  Let $i^{*}, j^{*}$ minimize $\inner{x_{i^{*}}}{x_{j^{*}}}$. If
  $\inner{v_{i^{*}}}{v_{j^{*}}} \le t$ then $\inner{x_{i^{*}}}{x_{j^{*}}} \le t+1$
  and the $i^{*}$-th entry is at least $\frac{1}{n+1}$; if $\inner{v_i}{v_j} \ge
  t+1$ for all $i \ne j$ then $\inner{x_i}{x_j} \ge t+2$ and the $i$-th entry is
  at most $\frac{\ell}{n^{2}} < \frac{1}{(n+1)^{1.5}}$. Again $\varphi_2$
  thresholds at $\frac{1}{n+1}$.
\end{proof}

\begin{theorem}[An attention unit solves MSD and LSD decision, Theorem C.2]
  \label{thm:attention-solves-msd-lsd}
  \uses{def:transformer, def:tf-solves, def:msd-dec, def:lsd-dec, def:attention-solves-predicate}
  An attention unit with input and output MLPs and parameters $d = \ell$,

\end{proof}

\begin{lemma}[Continuous thresholding MLP, Lemma D.1]
  \label{lem:mlp-threshold}
  \uses{lem:continuous-clipped-affine, lem:clipped-affine-values, lem:continuous-finite-product, lem:finite-product-zero-one}
  For any $a, b \in \R$ with $b > a$, there exists a continuous function
  $f : \R^{\ell} \to \R$ such that
  \[
    f(x) = \begin{cases}
      1 & \text{if } x[i] \ge b \text{ for some } 1 \le i \le \ell,\\
      0 & \text{if } x[i] < a \text{ for all } 1 \le i \le \ell.
    \end{cases}
  \]
  More formally, the existential witness consists of a function $f$, a proof
  that $f$ is continuous, a proof of
  $(\exists i,\ x[i]\ge b)\Rightarrow f(x)=1$, and a proof of
  $(\forall i,\ x[i]<a)\Rightarrow f(x)=0$. Proving only that some continuous
  function exists, for example the constant zero function, omits the behavioral
  guarantees and is not this lemma.
\end{lemma}
\begin{proof}
  \uses{lem:continuous-clipped-affine, lem:clipped-affine-values, lem:continuous-finite-product, lem:finite-product-zero-one, lem:continuity-closure}
  Define $g : \R \to \R$ by
  \[
    g(y)=\max\!\left(0,\min\!\left(1,\frac{y-a}{b-a}\right)\right).
  \]
  By Lemma~\ref{lem:continuous-clipped-affine}, $g$ is continuous, and by
  Lemma~\ref{lem:clipped-affine-values}, $g(y)=1$ when $y\ge b$ and $g(y)=0$
  when $y<a$. Let
  $f(x)=1-\prod_{i=1}^{\ell}(1-g(x[i]))$. Each coordinate map
  $x\mapsto 1-g(x[i])$ is continuous by composition with the coordinate
  projection and subtraction from a constant, so
  Lemma~\ref{lem:continuous-finite-product} gives continuity of
  $x\mapsto \prod_i(1-g(x[i]))$, and Lemma~\ref{lem:continuity-closure} gives
  continuity of $f$.

  If $x[i_0]\ge b$ for some $i_0$, then $g(x[i_0])=1$, so the $i_0$ factor
  $1-g(x[i_0])$ is $0$. Lemma~\ref{lem:finite-product-zero-one} gives that the
  whole product is $0$, hence $f(x)=1$. If $x[i]<a$ for all $i$, then every
  $g(x[i])=0$, so every factor is $1$; the all-one part of
  Lemma~\ref{lem:finite-product-zero-one} gives that the product is $1$, hence
  $f(x)=0$.

  In Lean, the continuity subproof should be organized as follows. First obtain
  `hg : Continuous g` from Lemma~\ref{lem:continuous-clipped-affine}. Then prove
  the product is continuous using `continuous_finsetProd Finset.univ`; for each
  coordinate $i$, the factor continuity is
  `continuous_const.sub (hg.comp (continuous_apply i))`. Finally close the
  continuity of $f$ by subtracting this product from `continuous_const`.
\end{proof}

\begin{lemma}[Continuous coordinate-appending MLP, Lemma D.2]
  \label{lem:mlp-append}
  \uses{lem:continuity-closure}
  For every $\tau>1$, there exists a continuous function
  $f : \R^{\ell} \to \R^{\ell+1}$ such that
  \[
    f(x) = \begin{cases}
      (x, 1) & \text{if } x[\ell] \le 1,\\
      (0, x) & \text{if } x[\ell] \ge \tau.
    \end{cases}
  \]
  The formal witness must include the continuity proof and both regional
  equalities:
  $x[\ell]\le1\Rightarrow f(x)=(x,1)$ and
  $x[\ell]\ge\tau\Rightarrow f(x)=(0,x)$. Equivalently, in coordinate form,
  for all $r\le\ell$:
  \[
    x[\ell]\le1 \Rightarrow
    \bigl(f(x)[r]=x[r]\ (r\le \ell)\ \wedge\ f(x)[\ell+1]=1\bigr),
  \]
  and
  \[
    x[\ell]\ge\tau \Rightarrow
    \bigl(f(x)[1]=0\ \wedge\ f(x)[r+1]=x[r]\ (1\le r\le\ell)\bigr).
  \]
  The proposition carried by this lemma is the conjunction
  \[
    \exists f,\ \operatorname{Continuous}(f)\ \wedge\
    \forall x,\ x[\ell]\le1\Rightarrow f(x)=(x,1)\ \wedge\
    \forall x,\ x[\ell]\ge\tau\Rightarrow f(x)=(0,x).
  \]
  The two implication conclusions are equalities of vectors, not `True'
  placeholders; a constant function cannot witness this lemma.
\end{lemma}
\begin{proof}
  \uses{lem:continuity-closure}
  Define $g : \R \to \R$ by $g(x) = 1$ if $x \le 1$,
  $g(x) = \frac{\tau-x}{\tau-1}$ if $1 < x < \tau$, and $g(x) = 0$ if
  $x \ge \tau$, which is continuous. Let $f_1(x) = (x, 1)$ and
  $f_2(x) = (0, x)$ in $\R^{\ell+1}$, both continuous. Set
  \[
    f(x) = g(x[\ell])\cdot f_1(x) + (1 - g(x[\ell]))\cdot f_2(x).
  \]
  By Lemma~\ref{lem:continuity-closure}, $f$ is continuous, and one checks
  directly that it equals $(x,1)$ when $x[\ell] \le 1$ and $(0,x)$ when
  $x[\ell] \ge \tau$ (and interpolates continuously in between).
\end{proof}

\begin{lemma}[Continuous normalizing MLP, Lemma D.3]
  \label{lem:mlp-normalize}
  \uses{lem:continuity-closure, lem:mlp-append}
  For every $\tau>1$ and $\rho>0$, there exists a continuous function
  $f : \R^{\ell} \to \R^{\ell+1}$ such that, for every $x$ with
  $\norm{x}\ge \rho$,
  \[
    f(x) = \begin{cases}
      \left(\frac{x}{\norm{x}}, 1\right) & \text{if } x[\ell] \le 1,\\
      \left(0, \frac{x}{\norm{x}}\right) & \text{if } x[\ell] \ge \tau.
    \end{cases}
  \]
  The formal witness must include continuity on all of $\R^\ell$ and the two
  displayed normalization equalities on the restricted domain $\norm{x}\ge\rho$.
  It must therefore construct a safe denominator or equivalent continuous
  extension. In full, the theorem asserts
  \[
    \exists f,\ \operatorname{Continuous}(f)\ \wedge\
    \forall x,\ \norm{x}\ge\rho \Rightarrow
      \left(x[\ell]\le1\Rightarrow
        f(x)=\left(\frac{x}{\norm{x}},1\right)\right)
      \wedge
      \left(x[\ell]\ge\tau\Rightarrow
        f(x)=\left(0,\frac{x}{\norm{x}}\right)\right).
  \]
  The norm lower bound, the two guard conditions, and the two vector equalities
  are all part of the statement. A proof that returns an arbitrary continuous
  function and proves only `True' under the norm hypothesis does not satisfy the
  blueprint node.
\end{lemma}
\begin{proof}
  \uses{lem:mlp-append, lem:continuity-closure, def:l2norm}
  Let $g$ be the same continuous cutoff as in Lemma~\ref{lem:mlp-append}. Define
  the safe denominator $s_\rho(x)=\max(\norm{x},\rho/2)$, which is continuous and
  strictly positive. Let
  $f_1(x)=\left(\frac{x}{s_\rho(x)},1\right)$ and
  $f_2(x)=\left(0,\frac{x}{s_\rho(x)}\right)$, both continuous on all of
  $\R^\ell$. Then
  \[
    f(x) = g(x[\ell])\cdot f_1(x) + (1 - g(x[\ell]))\cdot f_2(x)
  \]
  is continuous by Lemma~\ref{lem:continuity-closure}. On inputs with
  $\norm{x}\ge\rho$, the safe denominator equals $\norm{x}$; if in addition
  $x[\ell]\le 1$ then $g(x[\ell])=1$, and if $x[\ell]\ge\tau$ then
  $g(x[\ell])=0$. Hence the required piecewise values hold on the separated
  nonzero input region used by the construction.
\end{proof}
